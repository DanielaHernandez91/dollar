17  package dollar.execution.simple;
18  
19  import dollar.api.Scope;
20  import dollar.api.execution.DollarExecutor;
21  import dollar.internal.runtime.script.DollarScriptSupport;
22  import org.jetbrains.annotations.NotNull;
23  import org.jetbrains.annotations.Nullable;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.util.concurrent.Callable;
28  import java.util.concurrent.ExecutorService;
29  import java.util.concurrent.Executors;
30  import java.util.concurrent.ForkJoinPool;
31  import java.util.concurrent.Future;
32  import java.util.concurrent.FutureTask;
33  import java.util.concurrent.ScheduledExecutorService;
34  import java.util.concurrent.TimeUnit;
35  
36  import static dollar.internal.runtime.script.DollarScriptSupport.currentScope;
37  import static java.lang.Runtime.getRuntime;
38  import static java.util.concurrent.Executors.newFixedThreadPool;
39  
40  public class ScopeAwareDollarExecutor implements DollarExecutor {
41      @NotNull
42      private static final Logger log = LoggerFactory.getLogger(ScopeAwareDollarExecutor.class);
43  
44      @Nullable
45      private ForkJoinPool forkJoinPool;
46      @Nullable
47      private ExecutorService backgroundExecutor;
48      @Nullable
49      private ScheduledExecutorService scheduledExecutor;
50  
51  
52      public ScopeAwareDollarExecutor() {
53          getRuntime().addShutdownHook(new Thread(this::forceStop));
54          start();
55      }
56  
57      @Override
58      public void start() {
59          forkJoinPool = new ForkJoinPool(getRuntime().availableProcessors() * 8);
60          backgroundExecutor = newFixedThreadPool(getRuntime().availableProcessors());
61          scheduledExecutor = Executors.newScheduledThreadPool(getRuntime().availableProcessors());
62      }
63  
64      @Override
65      public void stop() {
66          if (backgroundExecutor != null) {
67              backgroundExecutor.shutdown();
68              backgroundExecutor = null;
69          }
70          if (forkJoinPool != null) {
71              forkJoinPool.shutdown();
72              forkJoinPool = null;
73          }
74          if (scheduledExecutor != null) {
75              scheduledExecutor.shutdown();
76              scheduledExecutor = null;
77          }
78      }
79  
80      @NotNull
81      @Override
82      public DollarExecutor copy() {
83          return this;
84      }
85  
86      @NotNull
87      @Override
88      public <T> Future<T> executeInBackground(@NotNull Callable<T> callable) {
89          assert backgroundExecutor != null;
90          log.debug("Background Execution");
91          return backgroundExecutor.submit(wrap(callable));
92      }
93  
94      @NotNull
95      @Override
96      public <T> Future<T> executeNow(@NotNull Callable<T> callable) {
97          log.debug("Immediate Execution");
98          final FutureTask<T> tFutureTask = new FutureTask<>(callable);
99          tFutureTask.run();
100         return tFutureTask;
101     }
102 
103     @Override
104     public void forceStop() {
105         if (backgroundExecutor != null) {
106             backgroundExecutor.shutdownNow();
107             backgroundExecutor = null;
108         }
109         if (forkJoinPool != null) {
110             forkJoinPool.shutdownNow();
111             forkJoinPool = null;
112         }
113         if (scheduledExecutor != null) {
114             scheduledExecutor.shutdownNow();
115             scheduledExecutor = null;
116         }
117     }
118 
119     @Override
120     public void restart() {
121         stop();
122         start();
123     }
124 
125     @NotNull
126     @Override
127     public Future<?> scheduleEvery(long millis, @NotNull Runnable runnable) {
128         assert scheduledExecutor != null;
129         return scheduledExecutor.scheduleAtFixedRate(wrap(runnable), millis, millis, TimeUnit.MILLISECONDS);
130     }
131 
132     @NotNull
133     @Override
134     public <T> Future<T> submit(@NotNull Callable<T> callable) {
135         assert forkJoinPool != null;
136         log.debug("Forking");
137         return forkJoinPool.submit(wrap(callable));
138     }
139 
140     private Runnable wrap(@NotNull Runnable runnable) {
141         Scope scope = currentScope();
142         return () -> DollarScriptSupport.inScope(true, scope, newScope -> {
143             runnable.run();
144             return null;
145         });
146     }
147 
148     private <T> Callable<T> wrap(@NotNull Callable<T> callable) {
149         Scope scope = currentScope();
150         return () -> DollarScriptSupport.inScope(true, scope, newScope -> callable.call());
151     }
152 }
