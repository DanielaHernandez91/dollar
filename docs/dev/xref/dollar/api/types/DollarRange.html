45      public DollarRange(@NotNull Value start, @NotNull Value finish) {
46          super();
47          Value startUnwrap;
48          Value finishUnwrap;
49  
50          if (start.compareTo(finish) < 0) {
51              startUnwrap = start.$unwrap();
52              finishUnwrap = finish.$unwrap();
53          } else {
54              startUnwrap = finish.$unwrap();
55              finishUnwrap = start.$unwrap();
56              reversed = true;
57  
58          }
59  
60          assert startUnwrap != null;
61          assert finishUnwrap != null;
62          range = Range.closed(startUnwrap, finishUnwrap);
63      }

66      public DollarRange(boolean lowerBounds,
67                         boolean upperBounds,
68                         boolean closedLeft,
69                         boolean closedRight, @Nullable Value lower, @Nullable Value upper) {
70          super();
71          Value lowerBound;
72          Value upperBound;
73          if ((lower != null) && (upper != null) && (lower.compareTo(upper) > 0)) {
74              lowerBound = upper;
75              upperBound = lower;
76              reversed = true;
77          } else {
78              lowerBound = lower;
79              upperBound = upper;
80          }
81          if (!lowerBounds && !upperBounds) {
82              range = Range.all();
83          } else if (!lowerBounds) {
84              if (closedRight) {
85                  range = Range.atMost(upperBound);
86              } else {
87                  range = Range.lessThan(upperBound);
88              }
89          } else if (!upperBounds) {
90              if (closedLeft) {
91                  range = Range.atLeast(lowerBound);
92              } else {
93                  range = Range.greaterThan(lowerBound);
94              }
95          } else if (closedLeft) {
96              if (closedRight) {
97                  range = Range.closed(lowerBound, upperBound);
98              } else {
99                  //openRight
100                 range = Range.closedOpen(lowerBound, upperBound);
101             }
102         } else if (!closedLeft) {
103             //openLeft
104             if (closedRight) {
105                 range = Range.openClosed(lowerBound, upperBound);
106             } else {
107                 //openRight
108                 if (lowerBound.equals(upperBound)) {
109                     throw new IllegalArgumentException("Cannot have an open range with lower bounds being the same as upper " +
110                                                                "bounds");
111                 } else {
112                     range = Range.open(lowerBound, upperBound);
113                 }
114             }
115         } else {
116             throw new IllegalStateException();
117         }
118     }

146         if (type.is(Type._LIST)) {
147             return DollarStatic.$(toVarList());
148         } else if (type.is(Type._MAP)) {
149             return DollarStatic.$(toJavaMap());

184     @NotNull
185     @Override
186     public Value $get(@NotNull Value key) {
187         if (key.integer()) {
188             long keyL;
189             if (key.toLong() < 0) {
190                 keyL = size() + key.toLong();
191             } else {
192                 keyL = key.toLong();
193             }
194             Value upper = range.upperEndpoint();
195             assert upper != null;
196             Value lower = range.lowerEndpoint();
197             assert lower != null;
198 
199             // (1..3) 2,2 [1..3] 1,3
200             if (reversed && !range.hasUpperBound()) {
201                 throw new DollarException("Attempted to get an element from an unbounded range offset from the upper bound " +
202                                                   "(reversed)");
203             }
204             if (!reversed && !range.hasLowerBound()) {
205                 throw new DollarException("Attempted to get an element from an unbounded range offset from the lower bound " +
206                                                   "(not reversed)");
207             }
208             if (range.upperBoundType().equals(BoundType.OPEN)) {
209                 upper = upper.$dec();
210             }
211 
212             if (range.lowerBoundType().equals(BoundType.OPEN)) {
213                 lower = lower.$inc();
214             }
215             if (upper.compareTo(lower) < 0) {
216                 throw new DollarException("Elements not available in an empty range");
217             }
218             if (upper.integer()) {
219                 if ((upper.toLong() == lower.toLong()) && (keyL == 0)) {
220                     return DollarFactory.fromValue(lower);
221                 }
222                 final long result = reversed ? (upper.toLong() - keyL) : (lower.toLong() + keyL);
223                 return DollarFactory.fromValue(result);
224             }
225             if (upper.decimal()) {
226                 if ((upper.toDouble() == lower.toDouble()) && (keyL == 0)) {
227                     return DollarFactory.fromValue(lower);
228                 }
229                 final double diff = reversed ? (upper.toDouble() - keyL) : (lower.toDouble() + keyL);
230                 return DollarFactory.fromValue(diff + Math.signum(diff));
231             }
232             if (upper.equals(lower) && (keyL == 0)) {
233                 return DollarFactory.fromValue(lower);
234             }
235             if (!reversed) {
236                 Value start = lower;
237                 for (long i = 0; i < keyL; i++) {
238                     start = start.$inc();
239                 }
240                 return start;
241             } else {
242                 Value finish = upper;
243                 for (long i = 0; i < keyL; i++) {
244                     finish = finish.$dec();
245                 }
246                 return finish;
247             }
248 
249         }
250         throw new DollarFailureException(ErrorType.INVALID_RANGE_OPERATION);
251     }

562             }

568     @Override
569     public int compareTo(@NotNull Value o) {
570         if ($containsValue(o).isTrue()) {
571             return 0;
572         }
573         if (lower().compareTo(o) < 0) {
574             return -1;
575         }
576         if (upper().compareTo(o) > 0) {
577             return 1;
578         }
579         throw new IllegalStateException();
580     }

582     @NotNull
583     private Value diff() {
584         Value upper = range.upperEndpoint();
585         assert upper != null;
586         Value lower = range.lowerEndpoint();
587         assert lower != null;
588 
589         // (1..3) 2,2 [1..3] 1,3
590         if (!range.hasUpperBound()) {
591             return INFINITY;
592         }
593         if (!range.hasLowerBound()) {
594             return INFINITY;
595         }
596         if (range.upperBoundType().equals(BoundType.OPEN)) {
597             upper = upper.$dec();
598         }
599 
600         if (range.lowerBoundType().equals(BoundType.OPEN)) {
601             lower = lower.$inc();
602         }
603         if (upper.compareTo(lower) < 0) {
604             return $void();
605         }
606         if (upper.integer()) {
607             if (upper.toLong() == lower.toLong()) {
608                 return DollarFactory.fromValue(1);
609             }
610             final long diff = reversed ? (lower.toLong() - upper.toLong()) : (upper.toLong() - lower.toLong());
611             return DollarFactory.fromValue(diff + (long) Math.signum(diff));
612         }
613         if (upper.decimal()) {
614             if (upper.toDouble() == lower.toDouble()) {
615                 return DollarFactory.fromValue(1.0);
616             }
617             final double diff = reversed ? (lower.toDouble() - upper.toDouble()) : (upper.toDouble() - lower.toDouble());
618             return DollarFactory.fromValue(diff + Math.signum(diff));
619         }
620         int count = 0;
621         Value start = lower;
622         Value finish = upper;
623         if (start.compareTo(finish) < 1) {
624             for (Value i = start; i.compareTo(finish) <= 0; i = i.$inc()) {
625                 count++;
626             }
627         } else {
628             for (Value i = start; i.compareTo(finish) <= 0; i = i.$dec()) {
629                 count--;
630             }
631         }
632         return DollarFactory.fromValue(count);
633 
634     }
