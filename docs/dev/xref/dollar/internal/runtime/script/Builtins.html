39      private static final HashMap<String, Builtin<?>> map;

42      static {
43          map = new HashMap<>();
44          addDollarStyle(1, 1, (pure, args, scope) -> args.get(0).$abs(), true, Type._ANY, "ABS");
45  
46          addJavaStyle(1, Integer.MAX_VALUE, (pure, args, scope) -> {
47              String message = args.get(0).$S();
48              ArrayList<var> values = new ArrayList<>(args);
49              values.remove(0);
50              return $(String.format(message, values.stream().map(var::toJavaObject).toArray()));
51          }, true, Type._STRING, "FORMAT");
52  
53          addDollarSingleNoScope(false, StateAware::$start, Type._VOID, "START");
54          addDollarSingleNoScope(false, StateAware::$stop, Type._VOID, "STOP");
55          addDollarSingleNoScope(false, StateAware::$create, Type._VOID, "CREATE");
56          addDollarSingleNoScope(false, StateAware::$destroy, Type._VOID, "DESTROY");
57          addDollarSingleNoScope(false, StateAware::$pause, Type._VOID, "PAUSE");
58          addDollarSingleNoScope(false, StateAware::$unpause, Type._VOID, "UNPAUSE");
59          addDollarSingleNoScope(false, StateAware::$state, Type._VOID, "STATE");
60  
61          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(
62                  0).toDouble() / DAY_IN_MILLIS), true, Type._DECIMAL, "MS", "MILLIS", "MILLISECONDS", "MILLISECOND");
63          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(
64                  0).toDouble() / DAY_IN_SECONDS), true, Type._DECIMAL, "S", "SEC", "SECS", "SECONDS");
65          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(
66                  0).toDouble() / DAY_IN_MINUTES), true, Type._DECIMAL, "M", "MINUTES",
67                       "MINUTE");
68          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(
69                  0).toDouble() / DAY_IN_HOURS), true, Type._DECIMAL, "H", "HOUR", "HOURS");
70          addJavaStyle(1, 1, (pure, args, scope) -> {
71              try {
72                  Thread.sleep((long) (args.get(0).toDouble() * DAY_IN_MILLIS));
73                  return args.get(0);
74              } catch (InterruptedException e) {
75                  return scope.handleError(e);
76              }
77          }, false, Type._VOID, "SLEEP");
78          addDollarStyle(1, 2, (pure, args, scope) -> {
79              if (args.size() == 1) {
80                  return DollarFactory.failure(ErrorType.valueOf(args.get(0).toString()), "", true);
81              } else {
82                  return DollarFactory.failure(ErrorType.valueOf(args.get(0).toString()), args.get(1).toString(), true);
83              }
84          }, true, Type._ANY, "ERROR");
85          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(0).toDouble()), true,
86                       Type._DECIMAL, "DAYS", "DAY", "D");
87          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(
88                  0).toDouble() * WEEK_IN_DAYS), true, Type._DECIMAL, "WEEKS", "WEEK", "W");
89          addJavaStyle(1, 1, (pure, args, scope) -> (args.get(0).toDouble() == null) ? args.get(0) : (args.get(

93          addJavaStyle(1, 1, (pure, args, scope) -> args.get(0).toString().length(), true, Type._INTEGER, "LEN");
94          addJavaStyle(0, 0, (pure, args, scope) -> $(new Date()), false, Type._DATE, "DATE");
95          addJavaStyle(0, 0, (pure, args, scope) -> System.currentTimeMillis(), false, Type._INTEGER, "TIME");
96          addJavaStyle(2, 2, (pure, args, scope) -> args.get(0).toString().matches(args.get(1).$S()), true, Type._BOOLEAN, "MATCHES");

102     private static void addDollarSingleNoScope(boolean isPure, @NotNull Function<var, var> lambda,
103                                                @NotNull Type type, @NotNull String... names) {
104         for (String name : names) {
105             map.put(name, new Builtin.BuiltinImpl(name, (Builtin<var>) (pure, args, scope) -> {
106                 var v = args.get(0);
107                 return lambda.apply(v);
108             }, 1, 1, isPure, type));
109         }
110     }
111 
112     private static void addDollarStyle(int minargs, int maxargs, @NotNull Builtin.DollarStyle lambda, boolean pure,
113                                        @NotNull Type type, @NotNull String... names) {
114         for (String name : names) {
115             map.put(name, new Builtin.BuiltinImpl(name, lambda, minargs, maxargs, pure, type));
116         }
117     }
118 
119     private static <T> void addJavaStyle(int minargs, int maxargs, @NotNull Builtin.JavaStyle<T> lambda, boolean pure,
120                                          @NotNull Type type, @NotNull String... names) {
121         for (String name : names) {
122             map.put(name, new Builtin.BuiltinImpl(name, lambda, minargs, maxargs, pure, type));
123         }
124     }
125 
126     @NotNull
127     public static var execute(@NotNull String name, @NotNull List<var> parameters, boolean pure) {
128         final Builtin<var> builtin = (Builtin<var>) map.get(name);
129         if (builtin == null) {
130             throw new BuiltinNotFoundException(name);
131         }
132         return builtin.execute(pure, parameters, DollarScriptSupport.currentScope());
133     }
134 
135     public static boolean exists(@NotNull String name) {
136         return map.containsKey(name);
137     }
138 
139     public static boolean isPure(@NotNull String lhsString) {
140         return (map.get(lhsString) != null) && ((Builtin.BuiltinImpl) map.get(lhsString)).isPure();
141     }
142 
143     @Nullable
144     public static Type type(@NotNull String s) {
145         return (((Builtin.BuiltinImpl<?>) map.get(s)).type());
146     }
