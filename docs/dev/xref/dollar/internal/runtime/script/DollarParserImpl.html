124     @Override
125     @NotNull
126     public var parse(@NotNull ScriptScope scope, @NotNull String source) throws Exception {
127         var v = DollarScriptSupport.inScope(false, scope, newScope -> {
128             DollarStatic.context().setClassLoader(classLoader);
129             Parser<?> parser = script();
130             try {
131                 parser.from(TOKENIZER, DollarLexer.IGNORED).parse(source);
132             } catch (RuntimeException e) {
133                 ErrorHandlerFactory.instance().handleTopLevel(e, null,
134                                                               (file != null) ? new File(file) : null);
135 
136             }
137             return $(exports);
138         });
139         if (v != null) {
140             return v;
141         } else {
142             throw new AssertionError("parse should not return null");
143         }
144 
145     }

150

191     @NotNull

193         PegDownProcessor pegDownProcessor = new PegDownProcessor(Extensions.FENCED_CODE_BLOCKS);
194         RootNode root =
195                 pegDownProcessor.parseMarkdown(asCharSource(file, Charset.forName("utf-8"))
196                                                        .read().toCharArray());
197         root.accept(new CodeExtractionVisitor());
198         return $();

201     @NotNull
202     private Parser<var> script() throws Exception {
203         log.debug("Starting Parse Phase");
204 
205 
206         Parser.Reference<var> ref = Parser.newReference();
207 //        Parser<var> block = block(ref.lazy(), false).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE));
208         Parser<var> expression = expression(false);
209         Parser<var> parser = (TERMINATOR_SYMBOL.optional(null)).next(expression.followedBy(
210                 TERMINATOR_SYMBOL).many1()).map(expressions -> {
211             log.debug("Ended Parse Phase");
212             log.debug("Starting Runtime Phase");
213             for (int i = 0; i < (expressions.size() - 1); i++) {
214                 expressions.get(i).$fixDeep(false);
215 //              System.err.println(fixed);
216             }
217             var resultVar = expressions.get(expressions.size() - 1);
218             var fixedResult = resultVar.$fixDeep(false);
219             log.debug("Ended Runtime Phase");
220             return fixedResult;
221         });
222         ref.set(parser);
223         return parser;
224 
225     }

228     @NotNull
229     private Parser<var> expression(final boolean pure) throws Exception {
230         Parser<var> main;
231         Parser.Reference<var> ref = Parser.newReference();
232         if (pure) {
233             main = ref.lazy()
234                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
235                            .or(Parsers.or(unitExpression(true),
236                                           listExpression(ref.lazy(), true),
237                                           mapExpression(ref.lazy(), true),
238                                           assertExpression(ref, true),
239                                           collectExpression(ref, true),
240                                           windowExpression(ref, true),
241                                           whenExpression(ref.lazy(), true),
242                                           functionCall(true),
243                                           rangeExpression(ref, true),
244                                           DECIMAL_LITERAL,
245                                           INTEGER_LITERAL,
246                                           STRING_LITERAL,
247                                           IDENTIFIER_KEYWORD,
248                                           variableRef(true),
249                                           builtin(true)))
250                            .or(blockExpression(ref.lazy(), true).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
251         } else {
252             main = ref.lazy()
253                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
254                            .or(Parsers.or(unitExpression(false),
255                                           listExpression(ref.lazy(), false),
256                                           mapExpression(ref.lazy(), false),
257                                           KEYWORD(PURE).next(expression(true)),
258                                           moduleExpression(ref),
259                                           assertExpression(ref, false),
260                                           collectExpression(ref, false),
261                                           windowExpression(ref, false),
262                                           whenExpression(ref.lazy(), false),
263                                           everyExpression(ref.lazy(), false),
264                                           functionCall(false),
265                                           scriptExpression(ref, false),
266                                           pureDefinitionOperator(ref),
267                                           rangeExpression(ref, false),
268                                           URL,
269                                           DECIMAL_LITERAL,
270                                           INTEGER_LITERAL,
271                                           STRING_LITERAL,
272                                           dollarIdentifier(ref),
273                                           IDENTIFIER_KEYWORD,
274                                           builtin(false),
275                                           variableRef(false)))
276                            .or(blockExpression(ref.lazy(), false).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
277         }
278 
279         OperatorTable<var> table = new OperatorTable<>();
280 
281         table = infixl(pure, table, PLUS, var::$plus);
282         table = infixl(pure, table, MINUS, var::$minus);
283         table = infixl(pure, table, CHOOSE, ControlFlowAware::$choose);
284         table = infixl(pure, table, DEFAULT, var::$default);
285         table = infixl(pure, table, DIVIDE, NumericAware::$divide);
286         table = infixl(pure, table, MOD, NumericAware::$modulus);
287 
288 
289         table = infixl(pure, table, INEQUALITY_OPERATOR, var::$notEquals);
290         table = infixl(pure, table, EQUALITY, var::$equals);
291         table = infixl(pure, table, AND, DollarStatic::$and);
292         table = infixl(pure, table, OR, DollarStatic::$or);
293         table = infixl(pure, table, LT, DollarStatic::$lt);
294         table = infixl(pure, table, GT, DollarStatic::$gt);
295         table = infixl(pure, table, LT_EQUALS, DollarStatic::$lte);
296         table = infixl(pure, table, GT_EQUALS, DollarStatic::$gte);
297         table = infixl(pure, table, MULTIPLY, Func::multiplyFunc);
298         table = infixl(pure, table, PAIR, Func::pairFunc);
299         table = infixl(pure, table, ELSE, Func::elseFunc);
300         table = infixl(pure, table, IN, Func::inFunc);
301         table = infixl(pure, table, WHEN_OP, Func::listenFunc);
302 
303         table = infixl(pure, table, EACH, (lhs, rhs) -> eachFunc(pure, lhs, rhs));
304         table = infixl(pure, table, REDUCE, (lhs, rhs) -> reduceFunc(pure, lhs, rhs));
305         table = infixl(pure, table, CAUSES, (lhs, rhs) -> causesFunc(pure, lhs, rhs));
306 
307 
308         table = infixlReactive(pure, table, ASSERT_EQ_REACT, Func::assertEqualsFunc);
309         table = infixlUnReactive(pure, table, ASSERT_EQ_UNREACT, Func::assertEqualsFunc);
310 
311 
312         table = postfix(pure, table, DEC, var::$dec);
313         table = postfix(pure, table, INC, var::$inc);
314 
315         table = prefix(pure, table, NEGATE, var::$negate);
316         table = prefix(pure, table, SIZE, var::$size);
317 
318         table = prefix(pure, table, NOT, DollarStatic::$not);
319         table = prefix(pure, table, ERROR, Func::errorFunc);
320         table = prefix(pure, table, TRUTHY, DollarStatic::$truthy);
321 
322         table = postfix(pure, table, MIN, v -> v.$min(currentScope().parallel()));
323         table = postfix(pure, table, MAX, v -> v.$max(currentScope().parallel()));
324         table = postfix(pure, table, SUM, v -> v.$sum(currentScope().parallel()));
325         table = postfix(pure, table, PRODUCT, v -> v.$product(currentScope().parallel()));
326         table = postfix(pure, table, SPLIT, var::$list);
327         table = prefix(pure, table, SORT, v -> v.$sort(currentScope().parallel()));
328         table = postfix(pure, table, REVERSE, var -> var.$reverse(currentScope().parallel()));
329         table = postfix(pure, table, UNIQUE, v -> v.$unique(currentScope().parallel()));
330         table = postfix(pure, table, AVG, v -> v.$avg(currentScope().parallel()));
331 
332         table = prefixUnReactive(pure, table, FIX, VarInternal::$fixDeep);
333 
334         table = table.prefix(parallelOperator(ref, pure), PARALLEL.priority());
335         table = table.prefix(serialOperator(ref, pure), SERIAL.priority());
336         //More complex expression syntax
337         table = table.postfix(pipeOperator(ref, pure), PIPE_OP.priority());
338         table = table.postfix(isOperator(pure), EQ_PRIORITY);
339         table = table.postfix(memberOperator(ref, pure), MEMBER.priority());
340         table = table.postfix(subscriptOperator(ref, pure), SUBSCRIPT_OP.priority());
341         table = table.postfix(castOperator(pure), CAST.priority());
342         table = table.postfix(parameterOperator(ref, pure), PARAM_OP.priority());
343 
344         table = table.prefix(ifOperator(ref, pure), IF_OP.priority());
345         table = table.prefix(forOperator(ref, pure), FOR_OP.priority());
346         table = table.prefix(whileOperator(ref, pure), WHILE_OP.priority());
347         table = table.prefix(variableUsageOperator(pure), 1000);
348 
349         table = table.prefix(assignmentOperator(ref, pure), ASSIGNMENT.priority());
350         table = table.prefix(definitionOperator(ref, pure), DEFINITION.priority());
351 
352         if (!pure) {
353             table = infixl(false, table, PUBLISH, Func::publishFunc);
354             table = infixl(false, table, SUBSCRIBE, Func::subscribeFunc);
355             table = infixl(false, table, WRITE_SIMPLE, Func::writeFunc);
356             table = prefix(false, table, READ_SIMPLE, Func::readFunc);
357 
358             table = prefix(false, table, DRAIN, URIAware::$drain);
359             table = prefix(false, table, ALL, URIAware::$all);
360             table = prefix(false, table, STOP, var::$stop);
361             table = prefix(false, table, START, var::$start);
362             table = prefix(false, table, PAUSE, var::$pause);
363             table = prefix(false, table, UNPAUSE, var::$unpause);
364             table = prefix(false, table, DESTROY, var::$destroy);
365             table = prefix(false, table, CREATE, var::$create);
366             table = prefix(false, table, STATE, var::$state);
367 
368             table = prefix(false, table, PRINT, DollarStatic::$out);
369             table = prefix(false, table, DEBUG, DollarStatic::$debug);
370             table = prefix(false, table, ERR, DollarStatic::$err);
371             table = prefix(false, table, FORK, Func::forkFunc);
372 
373             table = table.prefix(writeOperator(ref), WRITE_OP.priority());
374             table = table.prefix(readOperator(), READ_OP.priority());
375         }
376         Parser<var> parser = table.build(main);
377         ref.set(parser);
378         return parser;
379 
380     }

384                                        @NotNull OperatorTable<var> table,

393                                             prefix(DollarLexer.KEYWORDS.token(operator.keyword()).token().map(
394                                                     new SourceMapper<>(new DollarUnaryOperator(this,
395                                                                                                operator, f2, pure))),
396                                                    operator.priority());
397     }

449     private Parser<var> dollarIdentifier(@NotNull Parser.Reference<var> ref) {
450         return OP(DOLLAR).next(
451                 array(Terminals.Identifier.PARSER, OP(DEFAULT).next(ref.lazy()).optional(null)).between(
452                         OP(LEFT_BRACE),
453                         OP(RIGHT_BRACE)
454                 )
455         )
456                        .token()
457                        .map(token -> {
458                            Object[] objects = (Object[]) token.value();
459                            String varName = objects[0].toString();
460                            var defaultVal = (var) objects[1];
461                            return variableNode(false, varName, false, defaultVal, token, this);
462                        });
463     }

473     private Parser<var> unitExpression(boolean pure) {
474         return array(DECIMAL_LITERAL.or(INTEGER_LITERAL), BUILTIN)
475                        .token()
476                        .map(
477                                token -> {
478                                    Object[] objects = (Object[]) token.value();
479                                    var quantity = (var) objects[0];
480                                    var unit = (var) objects[1];
481                                    return node(UNIT_OP, pure, this, token, asList(quantity, unit),
482                                                i -> {
483                                                    String unitName = unit.toString();
484                                                    if (Builtins.exists(unitName)) {
485                                                        return Builtins.execute(unitName, singletonList(quantity), pure);
486                                                    } else {
487                                                        final var variable = variableNode(pure, unitName, token, this);
488                                                        currentScope().parameter("1", quantity);
489                                                        return DollarScriptSupport.fix(variable);
490                                                    }
491                                                });
492                                });
493     }

495     private Parser<var> listExpression(@NotNull Parser<var> expression, boolean pure) {
496         return OP_NL(LEFT_BRACKET)
497                        .next(expression.sepBy(COMMA_OR_NEWLINE_TERMINATOR))
498                        .followedBy(COMMA_OR_NEWLINE_TERMINATOR.optional(null))
499                        .followedBy(NL_OP(RIGHT_BRACKET))
500                        .token()
501                        .map(
502                                token -> {
503                                    List<var> entries = (List<var>) token.value();
504                                    final var node = node(LIST_OP, pure, this, token, entries,
505                                                          vars -> {
506                                                              log.info("Fixing list {}",
507                                                                       currentScope().parallel() ? "parallel" : "serial");
508                                                              return DollarFactory.fromList(entries).$fix(1,
509                                                                                                          currentScope().parallel());
510                                                          }
511                                    );
512                                    entries.forEach(entry -> entry.$listen(i -> node.$notify()));
513                                    return node;
514 
515                                });
516     }

518     private Parser<var> rangeExpression(@NotNull Parser.Reference<var> expression, boolean pure) {
519         return array(or(OP(LEFT_BRACKET), OP(LEFT_PAREN)),//0
520                      expression.lazy().optional(null),//1
521                      OP(RANGE),//2
522                      expression.lazy().optional(null),//3
523                      or(OP(RIGHT_BRACKET), OP(RIGHT_PAREN))//4
524         ).token()
525                        .map(
526                                token -> {
527                                    Object[] objects = (Object[]) token.value();
528                                    var upperBound = (var) objects[3];
529                                    var lowerBound = (var) objects[1];
530                                    return node(RANGE, pure, this, token, Arrays.asList(lowerBound, upperBound),
531                                                args -> {
532                                                    boolean closedLeft = objects[0].toString().equals(LEFT_BRACKET.symbol());
533                                                    boolean closedRight = objects[4].toString().equals(RIGHT_BRACKET.symbol());
534                                                    boolean lowerBounds = lowerBound != null;
535                                                    boolean upperBounds = upperBound != null;
536                                                    if (!closedLeft && !closedRight && (lowerBound != null) && (upperBound != null)
537                                                                && lowerBound.$unwrap().equals(upperBound.$unwrap())) {
538                                                        throw new DollarScriptException("Cannot create an open range with " +
539                                                                                                "identical upper and lower " +
540                                                                                                "bounds", new SourceSegmentValue
541                                                                                                                  (currentScope(),
542                                                                                                                   token));
543                                                    } else {
544                                                        return DollarFactory.wrap(
545                                                                new DollarRange(ImmutableList.of(), lowerBounds, upperBounds,
546                                                                                closedLeft, closedRight, lowerBound, upperBound));
547                                                    }
548                                                });
549 
550                                });
551     }

553     @NotNull
554     private Parser<var> mapExpression(@NotNull Parser<var> expression, boolean pure) {
555         return OP_NL(LEFT_BRACE)
556                        .next(expression.sepBy(COMMA_TERMINATOR))
557                        .followedBy(COMMA_TERMINATOR.optional(null))
558                        .followedBy(NL_OP(RIGHT_BRACE))
559                        .token()
560                        .map(token -> {
561                            List<var> o = (List<var>) token.value();
562                            final var node = node(MAP_OP, pure, this, token, o, i -> mapFunc(o));
563                            o.forEach(entry -> entry.$listen(i -> node.$notify()));
564                            return node;
565                        });
566     }

568     @NotNull
569     private Parser<var> blockExpression(@NotNull Parser<var> parentParser, boolean pure) throws Exception {
570         Parser.Reference<var> ref = Parser.newReference();
571         //Now we do the complex part, the following will only return the last value in the
572         //block when the block is evaluated, but it will trigger execution of the rest.
573         //This gives it functionality like a conventional function in imperative languages
574         Parser<var> or = (or(parentParser, parentParser.between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE))))
575                                  .sepBy1(SEMICOLON_TERMINATOR)
576                                  .followedBy(SEMICOLON_TERMINATOR.optional(null))
577                                  .map(var -> var)
578                                  .token()
579                                  .map(token -> node(BLOCK_OP, pure, this, token, (List<var>) token.value(),
580                                                     i -> blockFunc((List<var>) token.value())));
581         ref.set(or);
582         return or;
583     }

1021     private static class BacktickScriptMap implements TokenMap<String> {
1022         @Nullable
1023         @Override
1024         public String map(@NotNull Token token) {
1025             final Object val = token.value();
1026             if (val instanceof Tokens.Fragment) {
1027                 Tokens.Fragment c = (Tokens.Fragment) val;
1028                 if (!"backtick".equals(c.tag())) {
1029                     return null;
1030                 }
1031                 return c.text();
1032             } else {
1033                 return null;
1034             }
1035         }
1036 
1037         @NotNull
1038         @Override
1039         public String toString() {
1040             return "backtick";
1041         }
