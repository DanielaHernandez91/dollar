123         return OP(DOLLAR).next(
124                 array(Terminals.Identifier.PARSER, OP(DEFAULT).next(ref.lazy()).optional()).between(

127                 t -> {
128                     Object[] objects = (Object[]) t.value();
129                     return getVariable(pure, objects[0].toString(), false, (var) objects[1], t,
130                                        this);
131                 });

243     @NotNull
244     private Parser<var> script(@NotNull Scope scope, @NotNull boolean pure) throws Exception {
245         log.debug("Starting Parse Phase");
246 
247 
248         Parser.Reference<var> ref = Parser.newReference();
249         Parser<var> block = block(ref.lazy(), pure).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE));
250         Parser<var> expression = expression(null, pure);
251         Parser<var> parser = (TERMINATOR_SYMBOL.optional()).next(or(expression).followedBy(
252                 TERMINATOR_SYMBOL).many1()).map(l -> {
253             log.debug("Ended Parse Phase");
254             log.debug("Starting Runtime Phase");
255             for (int i = 0; i < l.size() - 1; i++) {
256                 var fixed = l.get(i)._fixDeep(false);
257 //                System.err.println(fixed);
258             }
259             var resultVar = l.get(l.size() - 1);
260             var fixedResult = resultVar._fixDeep(false);
261             log.debug("Ended Runtime Phase");
262             return fixedResult;
263         });
264         ref.set(parser);
265         return parser;
266 
267     }

269     @NotNull
270     private Parser<var> block(@NotNull Parser<var> parentParser, boolean pure) throws Exception {
271         Parser.Reference<var> ref = Parser.newReference();
272         //Now we do the complex part, the following will only return the last value in the
273         //block when the block is evaluated, but it will trigger execution of the rest.
274         //This gives it functionality like a conventional function in imperative languages
275         Parser<var> or = (or(parentParser, parentParser.between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE))))
276                                  .sepBy1(SEMICOLON_TERMINATOR)
277                                  .followedBy(SEMICOLON_TERMINATOR.optional())
278                                  .map(var -> var).token().map(new BlockOperator(this, pure));
279         ref.set(or);
280         return or;
281     }

283     @NotNull
284     private Parser<var> expression(@Nullable Parser.Reference<var> ref,
285                                    final boolean pure) throws Exception {
286         Parser<var> main;
287         if (ref == null) {
288             ref = Parser.newReference();
289         }
290         if (!pure) {
291             main = ref.lazy()
292                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
293                            .or(or(unitValue(pure), list(ref.lazy(), pure), map(ref.lazy(), pure),
294                                   pureDeclarationOperator(ref, pure),
295                                   KEYWORD(PURE).next(expression(null, true)),
296                                   moduleStatement(ref), assertOperator(ref, pure),
297                                   collectStatement(ref.lazy(), pure),
298                                   whenStatement(ref.lazy(), pure), everyStatement(ref.lazy(), pure),
299                                   functionCall(),
300                                   java(), URL, DECIMAL_LITERAL, INTEGER_LITERAL,
301                                   STRING_LITERAL,
302                                   dollarIdentifier(ref, pure), IDENTIFIER_KEYWORD,
303                                   builtin(pure),
304                                   variableRef(pure)))
305                            .or(block(ref.lazy(), pure).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
306         } else {
307             main = ref.lazy()
308                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
309                            .or(or(unitValue(pure),
310                                   list(ref.lazy(), pure),
311                                   map(ref.lazy(), pure),
312                                   assertOperator(ref, pure),
313                                   collectStatement(ref.lazy(), pure),
314                                   whenStatement(ref.lazy(), pure),
315                                   functionCall(),
316                                   DECIMAL_LITERAL,
317                                   INTEGER_LITERAL,
318                                   STRING_LITERAL,
319                                   IDENTIFIER_KEYWORD,
320                                   variableRef(pure),
321                                   builtin(pure)))
322                            .or(block(ref.lazy(), pure).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
323         }
324 
325         OperatorTable<var> table = new OperatorTable<var>();
326         table = table.infixl(op(INEQUALITY_OPERATOR, new BinaryOp(this, INEQUALITY_OPERATOR, Func::inequality)),
327                              INEQUALITY_OPERATOR.priority())
328                         .infixl(op(EQUALITY, new BinaryOp(this, EQUALITY, Func::equality)), EQUALITY.priority())
329                         .infixl(op(AND, new BinaryOp(this, AND, Func::and)), AND.priority())
330                         .infixl(op(OR, new BinaryOp(this, OR, Func::orFunc)), OR.priority())
331                         .postfix(pipeOperator(ref, pure), PIPE_OPERATOR.priority())
332                         .infixl(op(RANGE, new BinaryOp(this, RANGE, Func::range)), RANGE.priority())
333                         .infixl(op(LT, new BinaryOp(this, LT, Func::lt)), LT.priority())
334                         .infixl(op(GT, new BinaryOp(this, GT, Func::gt)), GT.priority())
335                         .infixl(op(LT_EQUALS, new BinaryOp(this, LT_EQUALS, Func::lte)), LT_EQUALS.priority())
336                         .infixl(op(GT_EQUALS, new BinaryOp(this, GT_EQUALS, Func::gte)), GT_EQUALS.priority())
337                         .infixl(op(MULTIPLY, new BinaryOp(this, MULTIPLY, Func::multiply)), MULTIPLY.priority())
338                         .infixl(op(DIVIDE, new BinaryOp(this, DIVIDE, NumericAware::$divide)), DIVIDE.priority())
339                         .infixl(op(MOD, new BinaryOp(this, MOD, NumericAware::$modulus)), MOD.priority())
340                         .infixl(op(PLUS, new BinaryOp(this, PLUS, var::$plus)), PLUS.priority())
341                         .infixl(op(NEGATE, new BinaryOp(this, NEGATE, var::$minus)), NEGATE.priority())
342                         .infixl(op(PAIR, new BinaryOp(this, PAIR, Func::pair)), PAIR.priority())
343                         .infixl(op(ASSERT_EQ_REACT, new BinaryOp(false, ASSERT_EQ_REACT, this, Func::assertEquals)),
344                                 ASSERT_EQ_REACT.priority())
345                         .infixl(op(ASSERT_EQ_UNREACT, new BinaryOp(true, ASSERT_EQ_UNREACT, this, Func::assertEquals)),
346                                 ASSERT_EQ_UNREACT.priority())
347                         .prefix(op(TRUTHY, new UnaryOp(this, TRUTHY, Func::truthy)), TRUTHY.priority())
348                         .prefix(op(SIZE, new UnaryOp(this, SIZE, var::$size)), SIZE.priority())
349                         .infixl(op(ELSE, new BinaryOp(this, ELSE, Func::elseFunc)), ELSE.priority())
350                         .prefix(ifOperator(ref), IF_OPERATOR.priority())
351                         .infixl(op(IN, new BinaryOp(this, IN, (lhs, rhs) -> rhs.$contains(lhs))), IN.priority())
352                         .prefix(op(ERROR, new UnaryOp(this, ERROR, Func::error)), ERROR.priority())
353                         .postfix(isOperator(), EQ_PRIORITY)
354                         .infixl(op(EACH, new BinaryOp(this, EACH, (lhs, rhs) -> Func.each(pure, lhs, rhs))),
355                                 EACH.priority())
356                         .infixl(op(REDUCE, new BinaryOp(this, REDUCE, (lhs, rhs) -> Func.reduce(pure, lhs, rhs))),
357                                 REDUCE.priority())
358                         .infixl(op(CAUSES, new CausesOperator(pure, this)), CAUSES.priority())
359                         .infixl(op(LISTEN, new BinaryOp(this, LISTEN, Func::listenFunc)), LISTEN.priority())
360                         .infixl(op(CHOOSE, new BinaryOp(this, CHOOSE, ControlFlowAware::$choose)), CHOOSE.priority())
361                         .infixl(op(DEFAULT, new BinaryOp(this, DEFAULT, var::$default)), DEFAULT.priority())
362                         .postfix(memberOperator(ref), MEMBER.priority())
363                         .prefix(op(NOT, new UnaryOp(this, NOT, Func::notFunc)), UNARY_PRIORITY)
364                         .postfix(op(DEC, new UnaryOp(this, DEC, var::$dec)), INC_DEC_PRIORITY)
365                         .postfix(op(INC, new UnaryOp(this, INC, var::$inc)), INC_DEC_PRIORITY)
366                         .prefix(op(NEGATE, new UnaryOp(this, NEGATE, var::$negate)), NEGATE.priority())
367                         .prefix(op(PARALLEL, new UnaryOp(true, Func::parallelFunc, PARALLEL, this)), PARALLEL.priority())
368                         .prefix(op(SERIAL, new UnaryOp(true, v -> v._fixDeep(false), SERIAL, this)), SERIAL.priority())
369                         .prefix(forOperator(ref, pure), FOR_OP.priority())
370                         .prefix(whileOperator(ref, pure), WHILE_OP.priority())
371                         .postfix(subscriptOperator(ref), SUBSCRIPT_OP.priority())
372                         .postfix(parameterOperator(ref, pure), PARAM_OP.priority())
373                         .prefix(op(FIX, new UnaryOp(true, VarInternal::_fixDeep, FIX, this)), FIX.priority())
374                         .postfix(castOperator(), CAST.priority())
375                         .prefix(variableUsageOperator(pure), 1000)
376                         .prefix(assignmentOperator(ref, pure), ASSIGNMENT.priority())
377                         .prefix(definitionOperator(ref, pure), DEFINITION.priority());
378 
379         if (!pure) {
380             table = table.prefix(op(PRINT, new UnaryOp(false, Func::outFunc, PRINT, this)), PRINT.priority())
381                             .prefix(op(DEBUG, new UnaryOp(false, Func::debugFunc, DEBUG, this)), DEBUG.priority())
382                             .prefix(op(ERR, new UnaryOp(false, Func::errFunc, ERR, this)), ERR.priority())
383                             .prefix(writeOperator(ref), WRITE_OP.priority())
384                             .prefix(readOperator(), READ_OP.priority())
385                             .prefix(op(STOP, new UnaryOp(this, STOP, var::$stop)), STOP.priority())
386                             .prefix(op(START, new UnaryOp(this, START, var::$start)), START.priority())
387                             .prefix(op(PAUSE, new UnaryOp(this, PAUSE, var::$pause)), PAUSE.priority())
388                             .prefix(op(UNPAUSE, new UnaryOp(this, UNPAUSE, var::$unpause)), UNPAUSE.priority())
389                             .prefix(op(DESTROY, new UnaryOp(this, DESTROY, var::$destroy)), DESTROY.priority())
390                             .prefix(op(CREATE, new UnaryOp(this, CREATE, var::$create)), CREATE.priority())
391                             .prefix(op(STATE, new UnaryOp(this, STATE, var::$state)), STATE.priority())
392                             .prefix(op(FORK, new UnaryOp(this, FORK, Func::fork)), FORK.priority())
393                             .infixl(op(PUBLISH, new BinaryOp(this, PUBLISH, Func::publishFunc)), PUBLISH.priority())
394                             .infixl(op(SUBSCRIBE, new BinaryOp(this, SUBSCRIBE, (lhs, rhs) -> Func.subscribeFunc(pure, lhs, rhs))),
395                                     SUBSCRIBE.priority())
396                             .infixl(op(WRITE_SIMPLE, new BinaryOp(this, WRITE_SIMPLE, Func::writeFunc)), WRITE_SIMPLE.priority())
397                             .prefix(op(READ_SIMPLE, new UnaryOp(this, READ_SIMPLE, Func::readFunc)), READ_SIMPLE.priority())
398                             .prefix(op(DRAIN, new UnaryOp(this, DRAIN, URIAware::$drain)), DRAIN.priority())
399                             .prefix(op(ALL, new UnaryOp(this, ALL, URIAware::$all)), ALL.priority());
400 
401         }
402         Parser<var> parser = table.build(main);
403         ref.set(parser);
404         return parser;
405 
406     }

436     private Parser<var> list(@NotNull Parser<var> expression, boolean pure) {
437         return OP_NL(LEFT_BRACKET)
438                        .next(expression.sepBy(COMMA_OR_NEWLINE_TERMINATOR))
439                        .followedBy(COMMA_OR_NEWLINE_TERMINATOR.optional())
440                        .followedBy(NL_OP(RIGHT_BRACKET)).token().map(
441                         new ListOperator(this, pure));
442     }

445         return OP_NL(LEFT_BRACE).next(expression.sepBy(COMMA_TERMINATOR))
446                        .followedBy(COMMA_TERMINATOR.optional())
447                        .followedBy(NL_OP(RIGHT_BRACE)).token()
448                        .map(new MapOperator(this, pure));
449     }

451     private Parser<var> moduleStatement(@NotNull Parser.Reference<var> ref) {
452         final Parser<Object[]> param = array(IDENTIFIER.followedBy(OP(ASSIGNMENT)), ref.lazy());
453 
454         final Parser<List<var>> parameters =
455                 KEYWORD(WITH).optional().next((param).map(objects -> {
456                     var result = (var) objects[1];
457                     result.setMetaAttribute(NAMED_PARAMETER_META_ATTR, objects[0].toString());
458                     return result;
459                 }).sepBy(OP(COMMA)).between(OP(LEFT_PAREN), OP(RIGHT_PAREN)));
460 
461         return array(KEYWORD(MODULE), STRING_LITERAL.or(URL),
462                      parameters.optional()).token().map(new ModuleOperator(this));
463 
464     }

466     private Parser<var> assertOperator(@NotNull Parser.Reference<var> ref, boolean pure) {
467         return OP(ASSERT).next(or(array(STRING_LITERAL.followedBy(OP(PAIR)), ref.lazy()),
468                                   array(OP(PAIR).optional(), ref.lazy()))
469                                        .token()
470                                        .map(new AssertOperator(this)));
471     }

473     private Parser<var> collectStatement(Parser<var> expression, boolean pure) {
474         return KEYWORD_NL(COLLECT)
475                        .next(
476                                array(expression,
477                                      KEYWORD(UNTIL).next(expression).optional(),
478                                      KEYWORD(UNLESS).next(expression).optional(),
479                                      expression)

481                        .token()
482                        .map(new CollectOperator(this, pure));
483     }

485     private Parser<var> whenStatement(Parser<var> expression, boolean pure) {

492     private Parser<var> everyStatement(Parser<var> expression, boolean pure) {
493         return KEYWORD_NL(EVERY)

497                                    KEYWORD(UNLESS).next(
498                                            expression).optional(null),
499                                    expression))

510     final Parser<var> java() {
511         return token(new TokenMap<String>() {
512             @Nullable
513             @Override
514             public String map(@NotNull Token token) {
515                 final Object val = token.value();
516                 if (val instanceof Tokens.Fragment) {
517                     Tokens.Fragment c = (Tokens.Fragment) val;
518                     if (!c.tag().equals("java")) {
519                         return null;
520                     }
521                     return c.text();
522                 } else {
523                     return null;
524                 }
525             }

528             @Override

646     private Parser<Map<? super var, ? extends var>> parameterOperator(@NotNull Parser.Reference<var> ref,
647                                                                       boolean pure) {
648         return OP(LEFT_PAREN).next(
649                 or(array(IDENTIFIER.followedBy(OP(ASSIGNMENT)), ref.lazy()),
650                    array(OP(ASSIGNMENT).optional(), ref.lazy())).map(
651                         objects -> {
652                             //Is it a named parameter
653                             if (objects[0] != null) {
654                                 //yes so let's add the name as metadata to the value
655                                 var result = (var) objects[1];
656                                 result.setMetaAttribute(NAMED_PARAMETER_META_ATTR,
657                                                         objects[0].toString());
658                                 return result;
659                             } else {
660                                 //no, just use the value
661                                 return (var) objects[1];
662                             }
663                         }).sepBy(COMMA_TERMINATOR)).followedBy(OP(RIGHT_PAREN)).token().map(
664                 new ParameterOperator(this, pure));
665     }

744     private class SourceMapper<T> implements Map<Token, T> {
745         @NotNull
746         private final T value;
747 
748 
749         SourceMapper(@NotNull T value) {
750             this.value = value;
751         }
752 
753         @NotNull
754         @Override
755         public T map(@NotNull Token token) {
756             if (value instanceof Operator) {
757                 ((Operator) value).setSource(new SourceSegmentValue(currentScope(), token));
758             }
759             return value;
760         }
