56      @NotNull

100     public SourceNode(@NotNull Pipeable lambda,
101                       @NotNull SourceSegment source,
102                       @NotNull List<var> inputs,
103                       @NotNull String name,
104                       @NotNull DollarParser parser,
105                       @NotNull SourceNodeOptions sourceNodeOptions,
106                       @NotNull String id,
107                       boolean pure, @NotNull OpDef operation) {
108         parallel = sourceNodeOptions.isParallel();
109         this.pure = pure;
110         this.operation = operation;
111 
112 
113         if (sourceNodeOptions.isScopeClosure()) {
114             this.lambda = vars -> {
115                 List<Scope> attachedScopes = new ArrayList<>(DollarScriptSupport.scopes());
116                 return DollarScriptSupport.node(name + "-closure", pure,
117                                                 new SourceNodeOptions(false, false, sourceNodeOptions.isParallel()),
118                                                 parser, source, inputs, vars2 -> {
119                             for (Scope scope : attachedScopes) {
120                                 DollarScriptSupport.pushScope(scope);
121                             }
122                             try {
123                                 return lambda.pipe(vars);
124                             } finally {
125                                 Collections.reverse(attachedScopes);
126                                 for (Scope scope : attachedScopes) {
127                                     DollarScriptSupport.popScope(scope);
128                                 }
129 
130                             }
131 
132                         }, operation);
133             };
134         } else {
135             this.lambda = lambda;
136         }
137         this.parser = parser;
138         newScope = sourceNodeOptions.isNewScope();
139         scopeClosure = sourceNodeOptions.isScopeClosure();
140         this.id = id;
141         this.inputs = inputs;
142         this.name = name;
143         this.source = source;
144         meta.put(OPERATION_NAME, name);
145         meta.put(OPERATION, operation);
146         if (!this.pure) {
147             meta.put(IMPURE, "true");
148         }
149         meta.put(ID, id);
150     }

153     @Nullable
154     @Override
155     public Object invoke(@NotNull Object proxy, @NotNull Method method, @Nullable Object[] args) throws Throwable {
156         try {
157             if (Objects.equals(method.getName(), "source")) {
158                 return source;
159             }
160             if (Objects.equals(method.getName(), "dynamic")) {
161                 return true;
162             }
163 
164             if (Objects.equals(method.getName(), "$constrain")) {
165                 if (args != null) {
166                     return _constrain((var) proxy, (var) args[0], String.valueOf(args[1]));
167                 }
168             }
169 
170             if (Objects.equals(method.getName(), "predictType")) {
171                 if (prediction == null) {
172                     prediction = typeLearner.predict(name, source, inputs);
173                 }
174                 return prediction;
175             }
176 
177             Object result = null;
178             //Some operations do not require a valid scope
179             if ((!method.getName().startsWith("$fix") && (nonScopeOperations.contains(
180                     method.getName()) || method.getDeclaringClass().equals(
181                     MetadataAware.class) || method.getDeclaringClass().equals(
182                     VarInternal.class)))) {
183                 //This method does not require a valid scope for execution
184                 try {
185                     result = invokeMain(proxy, method, args);
186                 } catch (Throwable throwable) {
187                     return currentScope().handleError(throwable, source);
188                 }
189             } else {
190                 //This method does require a scope
191                 Scope useScope;
192 
193                 Object attachedScopes = meta.get(SCOPES);
194                 if (attachedScopes != null) {
195                     List<Scope> scopes = new ArrayList<>((List<Scope>) attachedScopes);
196                     for (Scope scope : scopes) {
197                         DollarScriptSupport.pushScope(scope);
198                     }
199                 }
200                 try {
201                     if (newScope) {
202                         if (pure || currentScope().pure()) {
203                             useScope = new PureScope(currentScope(), currentScope().getSource(), name, currentScope().getFile());
204                         } else {
205                             if (currentScope().pure() && (operation.pure() != null) && !operation.pure()) {
206                                 throw new DollarScriptException("Attempted to create an impure scope within a pure scope " +
207                                                                         "(" + currentScope() + ") for " + name,
208                                                                 source);
209                             }
210                             useScope = new ScriptScope(currentScope(), name, false);
211                         }
212                     } else {
213                         useScope = currentScope();
214                     }
215 
216                     result = DollarScriptSupport.inScope(true, useScope, s -> {
217                         if (DollarStatic.getConfig().debugScope()) {
218                             log.info("EXE: {} {} for {}", name, method.getName(), source.getShortSourceMessage());
219                         }
220                         try {
221                             try {
222                                 stack.get().add(this);
223                                 return invokeMain(proxy, method, args);
224                             } finally {
225                                 stack.get().remove(stack.get().size() - 1);
226                             }
227 
228                         } catch (Throwable throwable) {
229                             return currentScope().handleError(throwable, source);
230                         }
231                     });
232                 } finally {
233 
234                     if (attachedScopes != null) {
235                         List<Scope> scopes = new ArrayList<>((List<Scope>) attachedScopes);
236                         Collections.reverse(scopes);
237                         for (Scope scope : scopes) {
238                             DollarScriptSupport.popScope(scope);
239                         }
240                     }
241                 }
242 
243             }
244 
245 
246             if (method.getName().startsWith("$fixDeep") && (result != null) && (result instanceof var)) {
247                 typeLearner.learn(name, source, inputs, ((var) result).$type());
248             }
249             return result;
250         } catch (Exception e) {
251             log.warn(e.getMessage(), e);
252             return parser.getErrorHandler().handle(currentScope(), source, e);
253         }
254     }

258     public Object invokeMain(@Nullable Object proxy, @NotNull Method method, @Nullable Object[] args) throws Throwable {
259         if (stack.get().size() > MAX_STACK_DEPTH) {
260             throw new LambdaRecursionException(stack.get().size());
261 
262         }
263         try {
264             if (proxy == null) {
265                 return null;
266             } else if (Objects.equals(Object.class, method.getDeclaringClass())) {
267                 String name = method.getName();
268                 if ("equals".equals(name)) {
269                     return execute().equals(args[0]);
270                 } else if ("hashCode".equals(name)) {
271                     return execute().hashCode();
272                 } else if ("toString".equals(name)) {
273                     return execute().toString();
274                 } else {
275                     throw new IllegalStateException(String.valueOf(method));
276                 }
277 
278             } else if ("$unwrap".equals(method.getName())) {
279                 return proxy;
280 //                return lambda.pipe(in).$unwrap();
281             } else if ("$fix".equals(method.getName())) {
282 
283                 if (args.length == 1) {
284                     return executePipe(DollarFactory.fromValue(args[0])).$fix((Boolean) args[0]);
285                 } else {
286                     return executePipe(DollarFactory.fromValue(args[1])).$fix((int) args[0], (Boolean) args[1]);
287 
288                 }
289 
290             } else if ("$fixDeep".equals(method.getName())) {
291 
292                 if ((args == null) || (args.length == 0)) {
293                     return executePipe(DollarFactory.fromValue(parallel)).$fixDeep();
294                 } else {
295                     return executePipe(DollarFactory.fromValue(args[0])).$fixDeep((Boolean) args[0]);
296                 }
297 
298 
299             } else if ("constraintLabel".equals(method.getName())) {
300                 return meta.get(CONSTRAINT_FINGERPRINT);
301             } else if ("metaAttribute".equals(method.getName()) && (((args != null) ? args.length : 0) == 1)) {
302                 return meta.get(args[0].toString());
303             } else if ("metaAttribute".equals(method.getName()) && (((args != null) ? args.length : 0) == 2)) {
304                 meta.put(String.valueOf(args[0]), String.valueOf(args[1]));
305                 return null;
306             } else if ("meta".equals(method.getName()) && (((args != null) ? args.length : 0) == 1)) {
307                 return meta.get(args[0]);
308             } else if ("meta".equals(method.getName()) && (((args != null) ? args.length : 0) == 2)) {
309                 meta.put(args[0], args[1]);
310                 return null;
311             } else if ("$listen".equals(method.getName())) {
312                 String listenerId = UUID.randomUUID().toString();
313                 if (args.length == 2) {
314                     listenerId = String.valueOf(args[1]);
315                 }
316                 listeners.put(listenerId, (Pipeable) args[0]);
317                 return $(listenerId);
318             } else if ("$notify".equals(method.getName())) {
319                 if (notifyStack.get().contains(this)) {
320                     //throw new IllegalStateException("Recursive notify loop detected");
321                     return proxy;
322                 }
323                 notifyStack.get().add(this);
324                 final var value = execute();
325                 for (Pipeable listener : listeners.values()) {
326                     listener.pipe(value);
327                 }
328                 notifyStack.get().remove(this);
329                 return proxy;
330             } else if ("$remove".equals(method.getName())) {
331                 //noinspection SuspiciousMethodCalls
332                 listeners.remove(args[0]);
333                 return args[0];
334             } else if ("hasErrors".equals(method.getName())) {
335                 return false;
336             } else if ("$copy".equals(method.getName()) || "copy".equals(method.getName())) {
337                 return proxy;
338             } else {
339 //            System.err.println(method);
340 
341                 var out = execute();
342                 if (out == null) {
343                     return null;
344                 }
345                 return method.invoke(out, args);
346             }
347         } catch (InvocationTargetException e) {
348             return currentScope().handleError(e, source);
349         }
350     }

352     @NotNull
353     public var executePipe(var var) throws Exception {
354         return lambda.pipe(var);
355     }

357     @NotNull
358     public var _constrain(@NotNull var source, @Nullable var constraint, @Nullable String constraintSource) {
359         if ((constraint == null) || (constraintSource == null)) {
360             return source;
361         }
362         String constraintFingerprint = (String) meta.get(CONSTRAINT_FINGERPRINT);
363         if ((constraintFingerprint == null) || constraintSource.equals(constraintFingerprint)) {
364             meta.put(CONSTRAINT_FINGERPRINT, constraintSource);
365             return source;
366         } else {
367             throw new ConstraintViolation(source, constraint, constraintFingerprint, constraintSource);
368         }
369     }
