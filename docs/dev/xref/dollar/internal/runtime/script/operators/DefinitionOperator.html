30  import java.util.ArrayList;
31  import java.util.Arrays;
32  
33  import static com.sillelien.dollar.api.DollarStatic.$;
34  import static com.sillelien.dollar.api.DollarStatic.$void;
35  import static dollar.internal.runtime.script.DollarScriptSupport.*;
36  
37  public class DefinitionOperator implements Map<Token, Map<? super var, ? extends var>> {
38      private final boolean pure;
39      private final DollarParser parser;
40  
41      public DefinitionOperator(boolean pure, DollarParser parser) {
42          this.pure = pure;
43          this.parser = parser;
44      }
45  
46      @Nullable
47      public Map<? super var, ? extends var> map(@NotNull Token token) {
48          Object[] objects = (Object[]) token.value();
49  //        final String constraintSource;
50  //        if (objects[1] instanceof var) {
51  //            constraintSource = ((var) objects[1])._source().getSourceSegment();
52  //        } else {
53  //            constraintSource = null;
54  //        }
55          Scope scope = currentScope();
56  
57          final Object exportObj;
58  
59          exportObj = objects[0];
60  
61          return new Map<var, var>() {
62              @NotNull
63              public var map(var v) {
64                  var value;
65                  Object variableNameObj;
66                  final Object typeConstraintObj;
67                  if (objects[2] != null && objects[2].toString().equals("def")) {
68                      variableNameObj = objects[3];
69                      if (objects.length == 5) {
70                          value = (var) objects[4];
71                      } else {
72                          value = v;
73                      }
74                      typeConstraintObj = objects[1];
75  
76                  } else {
77                      variableNameObj = objects[3];
78                      if (objects.length == 6) {
79                          value = (var) objects[5];
80                      } else {
81                          value = v;
82                      }
83                      typeConstraintObj = objects[2];
84  
85                  }
86                  var constraint;
87                  String constraintSource;
88                  if (typeConstraintObj != null) {
89                      constraint = DollarScriptSupport.node("definition-constraint", pure, SourceNodeOptions.NEW_SCOPE,
90                                                            token, new
91                                                                           ArrayList<>(),
92                                                            parser, i
93                                                                            -> {
94                                  final Type type = Type.valueOf(
95                                          typeConstraintObj.toString().toUpperCase());
96                                  var it = scope.getParameter("it");
97                                  return $(it.is(type));
98                              });
99                      constraintSource = typeConstraintObj.toString().toUpperCase();
100                 } else {
101                     constraint = null;
102                     constraintSource = null;
103                 }
104                 final String variableName = variableNameObj.toString();
105 
106                 var node = DollarScriptSupport.node(
107                         "assignment", pure, SourceNodeOptions.NO_SCOPE, parser, token, Arrays.asList(
108                                 constrain(scope, value, constraint, constraintSource)),
109                         args -> {
110                             setVariableDefinition(currentScope(), parser, token, pure, true,
111                                                   variableName,
112                                                   value,
113                                                   constraint,
114                                                   constraintSource
115                             );
116                             if (exportObj != null && exportObj.toString().equals("export")) {
117                                 parser.export(variableName, DollarScriptSupport.node(
118                                         "assignment", pure, SourceNodeOptions.NO_SCOPE, parser, token, Arrays.asList(value),
119                                         exportArgs -> value));
120                             }
121                             return $void();
122                         }
123                 );
124 
125                 node.$listen(i -> scope.notify(variableName));
126                 return node;
127             }
128         };
129     }
130 }
