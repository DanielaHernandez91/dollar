17  package dollar.internal.runtime.script.operators;
18  
19  import com.sillelien.dollar.api.var;
20  import dollar.internal.runtime.script.Builtins;
21  import dollar.internal.runtime.script.SourceNodeOptions;
22  import dollar.internal.runtime.script.api.DollarParser;
23  import org.jetbrains.annotations.NotNull;
24  import org.jparsec.Token;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.util.function.Function;
29  
30  import static dollar.internal.runtime.script.DollarScriptSupport.*;
31  import static dollar.internal.runtime.script.parser.Symbols.PIPE_OPERATOR;
32  import static java.util.Collections.singletonList;
33  
34  public class PipeOperator implements Function<Token, Function<var, var>> {
35  
36      @NotNull
37      private static final Logger log = LoggerFactory.getLogger("PipeOperator");
38      @NotNull
39      private final DollarParser parser;
40      private final boolean pure;
41  
42  
43      public PipeOperator(@NotNull DollarParser parser, boolean pure) {
44          assert PIPE_OPERATOR.validForPure(pure);
45  
46          this.parser = parser;
47          this.pure = pure;
48      }
49  
50  
51      @NotNull
52      @Override
53      public Function<var, var> apply(@NotNull Token token) {
54          var rhs = (var) token.value();
55          return lhs -> reactiveNode(
56                  "pipe", pure, SourceNodeOptions.NEW_SCOPE, parser, token, rhs,
57                  i -> inSubScope(false, pure, "pipe-runtime",
58                                  runtimeScope -> {
59                                      currentScope().setParameter("1", lhs);
60                                      var rhsVal = rhs._fix(false);
61                                      if ("function-call".equals(rhs.getMetaAttribute("operation"))) {
62                                          log.debug("PIPED TO FUNCTION");
63                                          return rhsVal;
64                                      } else {
65                                          String rhsStr = rhsVal.toString();
66                                          log.debug("OPERATION: {}", rhsStr);
67                                          if (rhs.getMetaAttribute("__builtin") != null) {
68                                              return Builtins.execute(rhsStr,
69                                                                      singletonList(lhs), pure);
70                                          } else {
71                                              //fixed to level 2 as we are executing the function
72                                              // in this scope
73                                              var variable = variableNode(pure, rhsStr, false, null, token,
74                                                                          parser);
75                                              assert variable != null;
76                                              return variable._fix(2, false);
77                                          }
78                                      }
79  
80                                  }));
81      }
82  }
