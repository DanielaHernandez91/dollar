117     private Parser<var> assertExpression(@NotNull Parser.Reference<var> ref, boolean pure) {
118         return OP(ASSERT)
119                        .next(
120                                or(
121                                        array(STRING_LITERAL.followedBy(OP(PAIR)), ref.lazy()),
122                                        array(OP(PAIR).optional(null), ref.lazy()
123                                        )
124                                )
125                                        .token()
126                                        .map(token -> {
127                                            Object[] objects = (Object[]) token.value();
128                                            var message = (var) objects[0];
129                                            var condition = (var) objects[1];
130                                            return util().reactiveNode(ASSERT, pure, condition, token, this,
131                                                                       i -> assertFunc(message, condition));
132                                        }));
133     }

135     private Parser<Function<? super var, ? extends var>> assignmentOperator(@NotNull Parser.Reference<var> ref,
136                                                                             boolean pure) {
137         assert ASSIGNMENT.validForPure(pure);
138 
139         return array(KEYWORD(EXPORT).optional(null),//0
140                      or(
141                              KEYWORD(CONST),
142                              KEYWORD(VOLATILE),
143                              KEYWORD(VAR)
144                      ).optional(null),//1
145                      IDENTIFIER.between(OP(LT), OP(GT)).optional(null),//2
146                      ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN)).optional(null).token().map((Token token) -> {
147                          if (token.value() != null) {
148                              ((var) token.value()).meta(CONSTRAINT_SOURCE, new SourceCode(util().scope(), token));
149                              return token.value();
150                          } else {
151                              return null;
152                          }
153                      }),//3
154                      OP(DOLLAR).next(ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN))).or(IDENTIFIER).or(BUILTIN), //4
155                      or(
156                              OP(ASSIGNMENT).map(i -> {
157                                  var var = $(i.toString());
158                                  var.metaAttribute(ASSIGNMENT_TYPE, "assign");
159                                  return var;
160                              }),
161                              OP(WHEN).next(ref.lazy()).followedBy(OP(ASSIGNMENT)).map(var -> {
162                                  var.metaAttribute(ASSIGNMENT_TYPE, "when");
163                                  return var;
164                              }),
165                              OP(SUBSCRIBE_ASSIGN).map(i -> {
166                                  var var = $(i.toString());
167                                  var.metaAttribute(ASSIGNMENT_TYPE, "subscribe");
168                                  return var;
169                              })
170                      )//5
171         ).token().map(new AssignmentOperator(pure, this));
172     }

174     @NotNull
175     private Parser<var> blockExpression(@NotNull Parser<var> parentParser, boolean pure) {
176         Parser.Reference<var> ref = Parser.newReference();
177         //Now we do the complex part, the following will only return the last value in the
178         //block when the block is evaluated, but it will trigger execution of the rest.
179         //This gives it functionality like a conventional function in imperative languages
180         Parser<var> or = (or(parentParser, parentParser.between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE))))
181                                  .sepBy1(SEMICOLON_TERMINATOR)
182                                  .followedBy(SEMICOLON_TERMINATOR.optional(null))
183                                  .map(var -> var)
184                                  .token()
185                                  .map(token -> util().node(BLOCK_OP,
186                                                            "block-" + util().shortHash(token), pure,
187                                                            this, token, (List<var>) token.value(),
188                                                            i -> blockFunc(Integer.MAX_VALUE, (List<var>) token.value())));
189         ref.set(or);
190         return or;
191     }

200     private Parser<Function<? super var, ? extends var>> castOperator(boolean pure) {
201         return KEYWORD(AS)
202                        .next(IDENTIFIER)
203                        .token()
204                        .map(token -> lhs -> {
205                                 assert CAST.validForPure(pure);
206                                 return util().reactiveNode(CAST, "cast-" + token.toString().toLowerCase(), pure, token, lhs,
207                                                            (var) token.value(), this,
208                                                            i -> castFunc(lhs, token.toString())
209                                 );
210                             }
211                        );
212     }

214     private Parser<var> classExpression(@NotNull Parser.Reference<var> ref) {
215         return KEYWORD_NL(CLASS_OP)
216                        .next(
217                                array(IDENTIFIER,
218                                      OP_NL(LEFT_BRACE).next(
219                                              (ref.lazy().prefix(
220                                                      or(assignmentOperator(ref, false),
221                                                         definitionOperator(ref, false)
222                                                      )
223                                              )).sepBy1(SEMICOLON_TERMINATOR).followedBy(SEMICOLON_TERMINATOR.optional(null))
224                                      ).followedBy(OP(RIGHT_BRACE))
225                                )
226                        )
227                        .token()
228                        .map(token -> {
229                            Object[] objects = (Object[]) token.value();
230                            ArrayList<var> in = new ArrayList<>();
231                            in.add((var) objects[0]);
232                            in.addAll((List<var>) objects[1]);
233                            return util().node(CLASS_OP, "class-" + objects[0],
234                                               false, this, token, in,
235                                               i -> {
236                                                   ClassScopeFactory factory = new ClassScopeFactory(util().scope(),
237                                                                                                     objects[0].toString(),
238                                                                                                     (List<var>) objects[1], false,
239                                                                                                     false);
240 
241                                                   util().scope().registerClass(objects[0].toString(), factory);
242                                                   return $void();
243                                               });
244                        });
245     }

259     private Parser<Function<? super var, ? extends var>> definitionOperator(@NotNull Parser.Reference<var> ref,
260                                                                             boolean pure) {
261 
262         return or(
263                 array(
264                         KEYWORD(EXPORT).optional(null),//0
265                         KEYWORD(CONST).optional(null), //1
266                         IDENTIFIER.between(OP(LT), OP(GT)).optional(null),//2
267                         (OP(DOLLAR).next(ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN))).or(IDENTIFIER))
268                                 .token().map((Token token) -> {
269                             ((var) token.value()).meta(CONSTRAINT_SOURCE, new SourceCode(util().scope(), token));
270                             return token.value();
271                         }), //3
272                         OP(DEFINITION) //4
273 
274                 ).token().map(new DefinitionOperator(pure, this, false)),
275                 array(
276                         KEYWORD(EXPORT).optional(null), //0
277                         IDENTIFIER.between(OP(LT), OP(GT)).optional(null), //1
278                         KEYWORD(DEF), //2
279                         IDENTIFIER //3
280 
281                 ).token().map(new DefinitionOperator(pure, this, true))
282 
283         );
284     }

286     private Parser<var> dollarIdentifier(@NotNull Parser.Reference<var> ref) {
287         return OP(DOLLAR).next(
288                 array(Terminals.Identifier.PARSER, OP(DEFAULT).next(ref.lazy()).optional(null)).between(
289                         OP(LEFT_BRACE),
290                         OP(RIGHT_BRACE)
291                 )
292         )
293                        .token()
294                        .map(token -> {
295                            Object[] objects = (Object[]) token.value();
296                            var defaultVal = (var) objects[1];
297                            return util().variableNode(false, VarKey.of(objects[0].toString()), false, defaultVal, token, this);
298                        });
299     }

301     private Parser<var> everyExpression(@NotNull Parser<var> expression, boolean pure) {
302         return KEYWORD_NL(EVERY_OP)
303                        .next(array(unitExpression(false),
304                                    KEYWORD(UNTIL).next(expression).optional(null),
305                                    KEYWORD(UNLESS).next(expression).optional(null),
306                                    expression))
307                        .token()
308                        .map(token -> {
309                            assert EVERY_OP.validForPure(pure);
310                            final AtomicInteger count = new AtomicInteger(-1);
311                            Object[] objects = (Object[]) token.value();
312                            var durationVar = (var) objects[0];
313                            var until = (var) objects[1];
314                            var unless = (var) objects[2];
315                            var block = (var) objects[3];
316                            return util().reactiveNode(EVERY_OP, false, block, token, this,
317                                                       i -> everyFunc(count, durationVar, until, unless, block));
318 
319                        });
320     }

334     @Override
335     @NotNull
336     public var parse(@NotNull Scope scope, @NotNull String source) {
337         DollarStatic.context().parser(this);
338         var v = util().inScope(false, scope, newScope -> {
339             DollarStatic.context().classLoader(classLoader);
340             Parser<?> parser = script();
341             try {
342                 parser.from(TOKENIZER, IGNORED).parse(source);
343             } catch (RuntimeException e) {
344                 ErrorHandlerFactory.instance().handleTopLevel(e, null, (file != null) ? new File(file) : null);
345 
346             }
347             HashMap<var, var> exportMap = new HashMap<>();
348             exports.forEach((varKey, var) -> exportMap.put(DollarFactory.fromStringValue(varKey.asString()), var));
349             return $(exportMap);
350         }).orElseThrow(() -> new AssertionError("Optional should not be null here"));
351         if (v != null) {
352             return v;
353         } else {
354             throw new AssertionError("parse should not return null");
355         }
356 
357     }

411     @NotNull
412     private Parser<var> expressionInternal(boolean pure, @NotNull Parser.Reference<var> ref) {
413         Parser<var> main;
414         if (pure) {
415             main = ref.lazy()
416                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
417                            .or(Parsers.or(unitExpression(true),
418                                           listExpression(ref.lazy(), true),
419                                           mapExpression(ref.lazy(), true),
420                                           assertExpression(ref, true),
421                                           collectExpression(ref, true),
422                                           windowExpression(ref, true),
423                                           whenExpression(ref.lazy(), true),
424                                           functionCall(true),
425                                           rangeExpression(ref, true),
426                                           variableUsageOperator(ref, true),
427                                           DECIMAL_LITERAL,
428                                           INTEGER_LITERAL,
429                                           STRING_LITERAL,
430                                           IDENTIFIER_KEYWORD,
431                                           variableRef(true),
432 
433                                           builtin(true)))
434                            .or(blockExpression(ref.lazy(), true).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
435         } else {
436             main = ref.lazy()
437                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
438                            .or(Parsers.or(unitExpression(false),
439                                           listExpression(ref.lazy(), false),
440                                           mapExpression(ref.lazy(), false),
441                                           KEYWORD(PURE).next(expression(true)),
442                                           classExpression(ref),
443                                           newExpression(ref, false),
444                                           thisRef(false),
445 //                                          printExpression(ref, false),
446                                           moduleExpression(ref),
447                                           assertExpression(ref, false),
448                                           collectExpression(ref, false),
449                                           windowExpression(ref, false),
450                                           whenExpression(ref.lazy(), false),
451                                           everyExpression(ref.lazy(), false),
452                                           functionCall(false),
453                                           scriptExpression(ref, false),
454                                           pureDefinitionOperator(ref),
455                                           rangeExpression(ref, false),
456                                           forkExpression(ref, false),
457                                           variableUsageOperator(ref, false),
458                                           URL,
459                                           DECIMAL_LITERAL,
460                                           INTEGER_LITERAL,
461                                           STRING_LITERAL,
462                                           dollarIdentifier(ref),
463                                           IDENTIFIER_KEYWORD,
464                                           builtin(false),
465                                           variableRef(false)))
466                            .or(blockExpression(ref.lazy(), false).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
467         }
468 
469         OperatorTable<var> table = new OperatorTable<>();
470 
471         table = infixl(pure, table, PLUS, var::$plus);
472         table = infixl(pure, table, MINUS, var::$minus);
473         table = infixl(pure, table, CHOOSE, var::$choose);
474         table = infixl(pure, table, DEFAULT, var::$default);
475         table = infixl(pure, table, DIVIDE, var::$divide);
476         table = infixl(pure, table, MOD, var::$modulus);
477 
478 
479         table = infixl(pure, table, INEQUALITY_OPERATOR, var::$notEquals);
480         table = infixl(pure, table, EQUALITY, var::$equals);
481         table = infixl(pure, table, AND, DollarStatic::$and);
482         table = infixl(pure, table, OR, DollarStatic::$or);
483         table = infixl(pure, table, LT, DollarStatic::$lt);
484         table = infixl(pure, table, GT, DollarStatic::$gt);
485         table = infixl(pure, table, LT_EQUALS, DollarStatic::$lte);
486         table = infixl(pure, table, GT_EQUALS, DollarStatic::$gte);
487         table = infixl(pure, table, MULTIPLY, Func::multiplyFunc);
488         table = infixl(pure, table, PAIR, Func::pairFunc);
489         table = infixl(pure, table, ELSE, Func::elseFunc);
490         table = infixl(pure, table, IN, Func::inFunc);
491         table = infixl(pure, table, WHEN, Func::listenFunc);
492 
493         table = infixl(pure, table, EACH, (lhs, rhs) -> eachFunc(pure, lhs, rhs));
494         table = infixl(pure, table, REDUCE, (lhs, rhs) -> reduceFunc(pure, lhs, rhs));
495         table = infixl(pure, table, CAUSES, (lhs, rhs) -> causesFunc(pure, lhs, rhs));
496 
497 
498         table = infixlReactive(pure, table, ASSERT_EQ_REACT, Func::assertEqualsFunc);
499         table = infixlUnReactive(pure, table, ASSERT_EQ_UNREACT, Func::assertEqualsFunc);
500 
501 
502         table = postfix(pure, table, DEC, (v, u) -> v.$dec());
503         table = postfix(pure, table, INC, (v, u) -> v.$inc());
504 
505         table = prefix(pure, table, NEGATE, (v, u) -> v.$negate());
506         table = prefix(pure, table, SIZE, (v, u) -> v.$size());
507 
508         table = prefix(pure, table, NOT, (v, u) -> DollarStatic.$not(v));
509         table = prefix(pure, table, ERROR, (v, u) -> Func.errorFunc(v));
510         table = prefix(pure, table, TRUTHY, (v, u) -> DollarStatic.$truthy(v));
511 
512         table = postfix(pure, table, MIN, (v, u) -> v.$min(false));
513         table = postfix(pure, table, MAX, (v, u) -> v.$max(false));
514         table = postfix(pure, table, SUM, (v, u) -> v.$sum(false));
515         table = postfix(pure, table, PRODUCT, (v, u) -> v.$product(false));
516         table = postfix(pure, table, SPLIT, (v, u) -> v.$list());
517         table = prefix(pure, table, SORT, (v, u) -> v.$sort(false));
518         table = postfix(pure, table, REVERSE, (v, u) -> v.$reverse(false));
519         table = postfix(pure, table, UNIQUE, (v, u) -> v.$unique(false));
520         table = postfix(pure, table, AVG, (v, u) -> v.$avg(false));
521 
522         table = prefixUnReactive(pure, table, FIX, (v1, v12) -> Func.fixFunc(v1));
523 
524         table = table.prefix(parallelOperator(ref, pure), PARALLEL.priority());
525         table = table.prefix(serialOperator(ref, pure), SERIAL.priority());
526 
527         //More complex expression syntax
528         table = table.postfix(pipeOperator(ref, pure), PIPE_OP.priority());
529         table = table.postfix(isOperator(pure), EQ_PRIORITY);
530         table = table.postfix(memberOperator(ref, pure), MEMBER.priority());
531         table = table.postfix(subscriptOperator(ref, pure), SUBSCRIPT_OP.priority());
532         table = table.postfix(castOperator(pure), CAST.priority());
533         table = table.postfix(parameterOperator(ref, pure), PARAM_OP.priority());
534 
535         table = table.prefix(ifOperator(ref, pure), IF_OP.priority());
536         table = table.prefix(forOperator(ref, pure), FOR_OP.priority());
537         table = table.prefix(whileOperator(ref, pure), WHILE_OP.priority());
538 
539 
540         table = table.prefix(assignmentOperator(ref, pure), ASSIGNMENT.priority());
541         table = table.prefix(definitionOperator(ref, pure), DEFINITION.priority());
542 
543         if (!pure) {
544             table = infixl(false, table, PUBLISH, Func::publishFunc);
545             table = infixl(false, table, SUBSCRIBE, Func::subscribeFunc);
546             table = infixl(false, table, WRITE_SIMPLE, Func::writeFunc);
547             table = prefix(false, table, READ_SIMPLE, (v, u) -> Func.readFunc(v));
548 
549             table = prefix(false, table, DRAIN, (v, u) -> v.$drain());
550             table = prefix(false, table, ALL, (v, u) -> v.$all());
551             table = prefix(false, table, STOP, (v, u) -> v.$stop());
552             table = prefix(false, table, START, (v, u) -> v.$start());
553             table = prefix(false, table, PAUSE, (v, u) -> v.$pause());
554             table = prefix(false, table, UNPAUSE, (v, u) -> v.$unpause());
555             table = prefix(false, table, DESTROY, (v, u) -> v.$destroy());
556             table = prefix(false, table, CREATE, (v, u) -> v.$create());
557             table = prefix(false, table, STATE, (v, u) -> v.$state());
558 
559             table = prefix(false, table, OUT, (v, u) -> printFunc(this, u, OUT, Arrays.asList(v)));
560             table = prefix(false, table, DEBUG, (v, u) -> printFunc(this, u, DEBUG, Arrays.asList(v)));
561             table = prefix(false, table, ERR, (v, u) -> printFunc(this, u, ERR, Arrays.asList(v)));
562 
563             table = table.prefix(writeOperator(ref), WRITE_OP.priority());
564             table = table.prefix(readOperator(), READ_OP.priority());
565         }
566         Parser<var> parser = table.build(main);
567         ref.set(parser);
568         return parser;
569     }
