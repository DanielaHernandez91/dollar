120     private Parser<Value> assertExpression(@NotNull Parser.Reference<Value> ref, boolean pure) {
121         return OP(ASSERT)
122                        .next(
123                                or(
124                                        array(STRING_LITERAL.followedBy(OP(PAIR)), ref.lazy()),
125                                        array(OP(PAIR).optional(null), ref.lazy()
126                                        )
127                                )
128                                        .token()
129                                        .map(token -> {
130                                            Object[] objects = (Object[]) token.value();
131                                            Value message = (Value) objects[0];
132                                            Value condition = (Value) objects[1];
133                                            return util().reactiveNode(ASSERT, pure, condition, token, this,
134                                                                       i -> assertFunc(message, condition));
135                                        }));
136     }

138     private Parser<Function<? super Value, ? extends Value>> assignmentOperator(@NotNull Parser.Reference<Value> ref,
139                                                                                 boolean pure) {
140         assert ASSIGNMENT.validForPure(pure);
141 
142         return array(KEYWORD(EXPORT).optional(null),//0
143                      or(
144                              KEYWORD(CONST),
145                              KEYWORD(VOLATILE),
146                              KEYWORD(VAR)
147                      ).optional(null),//1
148                      IDENTIFIER.between(OP(LT), OP(GT)).optional(null),//2
149                      ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN)).optional(null).token().map((Token token) -> {
150                          if (token.value() != null) {
151                              ((Value) token.value()).meta(CONSTRAINT_SOURCE, new SourceImpl(util().scope(), token));
152                              return token.value();
153                          } else {
154                              return null;
155                          }
156                      }),//3
157                      OP(DOLLAR).next(ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN))).or(IDENTIFIER).or(BUILTIN), //4
158                      or(
159                              OP(ASSIGNMENT).map(i -> {
160                                  Value Value = $(i.toString());
161                                  Value.metaAttribute(ASSIGNMENT_TYPE, "assign");
162                                  return Value;
163                              }),
164                              OP(WHEN).next(ref.lazy()).followedBy(OP(ASSIGNMENT)).map(var -> {
165                                  var.metaAttribute(ASSIGNMENT_TYPE, "when");
166                                  return var;
167                              }),
168                              OP(SUBSCRIBE_ASSIGN).map(i -> {
169                                  Value Value = $(i.toString());
170                                  Value.metaAttribute(ASSIGNMENT_TYPE, "subscribe");
171                                  return Value;
172                              })
173                      )//5
174         ).token().map(new AssignmentOperator(pure, this));
175     }

177     @NotNull
178     private Parser<Value> blockExpression(@NotNull Parser<Value> parentParser, boolean pure) {
179         Parser.Reference<Value> ref = Parser.newReference();
180         //Now we do the complex part, the following will only return the last value in the
181         //block when the block is evaluated, but it will trigger execution of the rest.
182         //This gives it functionality like a conventional function in imperative languages
183         Parser<Value> or = (or(parentParser, parentParser.between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE))))
184                                    .sepBy1(SEMICOLON_TERMINATOR)
185                                    .followedBy(SEMICOLON_TERMINATOR.optional(null))
186                                    .map(var -> var)
187                                    .token()
188                                    .map(token -> util().node(BLOCK_OP,
189                                                              "block-" + util().shortHash(token), pure,
190                                                              this, token, (List<Value>) token.value(),
191                                                              i -> blockFunc(Integer.MAX_VALUE, (List<Value>) token.value())));
192         ref.set(or);
193         return or;
194     }

203     private Parser<Function<? super Value, ? extends Value>> castOperator(boolean pure) {
204         return KEYWORD(AS)
205                        .next(IDENTIFIER)

207                        .map(token -> lhs -> {
208                                 assert CAST.validForPure(pure);
209                                 return util().reactiveNode(CAST, "cast-" + token.toString().toLowerCase(), pure, token, lhs,
210                                                            (Value) token.value(), this,
211                                                            i -> castFunc(lhs, token.toString())
212                                 );
213                             }
214                        );
215     }

217     private Parser<Value> classExpression(@NotNull Parser.Reference<Value> ref) {
218         return KEYWORD_NL(CLASS_OP)
219                        .next(
220                                array(IDENTIFIER,
221                                      OP_NL(LEFT_BRACE).next(
222                                              (ref.lazy().prefix(
223                                                      or(assignmentOperator(ref, false),
224                                                         definitionOperator(ref, false)
225                                                      )
226                                              )).sepBy1(SEMICOLON_TERMINATOR).followedBy(SEMICOLON_TERMINATOR.optional(null))
227                                      ).followedBy(OP(RIGHT_BRACE))
228                                )
229                        )
230                        .token()
231                        .map(token -> {
232                            Object[] objects = (Object[]) token.value();
233                            ArrayList<Value> in = new ArrayList<>();
234                            in.add((Value) objects[0]);
235                            in.addAll((List<Value>) objects[1]);
236                            return util().node(CLASS_OP, "class-" + objects[0],
237                                               false, this, token, in,
238                                               i -> {
239                                                   ClassScopeFactory factory = new ClassScopeFactory(util().scope(),
240                                                                                                     objects[0].toString(),
241                                                                                                     (List<Value>) objects[1], false,
242                                                                                                     false);
243 
244                                                   util().scope().registerClass(ClassName.of(objects[0]), factory);
245                                                   return $void();
246                                               });
247                        });
248     }

251         return KEYWORD_NL(COLLECT_OP)

254                                      KEYWORD(UNTIL).next(ref.lazy()).optional(null),
255                                      KEYWORD(UNLESS).next(ref.lazy()).optional(null),

262     private Parser<Function<? super Value, ? extends Value>> definitionOperator(@NotNull Parser.Reference<Value> ref,
263                                                                                 boolean pure) {
264 
265         return or(
266                 array(
267                         KEYWORD(EXPORT).optional(null),//0
268                         KEYWORD(CONST).optional(null), //1
269                         IDENTIFIER.between(OP(LT), OP(GT)).optional(null),//2
270                         (OP(DOLLAR).next(ref.lazy().between(OP(LEFT_PAREN), OP(RIGHT_PAREN))).or(IDENTIFIER))
271                                 .token().map((Token token) -> {
272                             ((Value) token.value()).meta(CONSTRAINT_SOURCE, new SourceImpl(util().scope(), token));
273                             return token.value();
274                         }), //3
275                         OP(DEFINITION) //4
276 
277                 ).token().map(new DefinitionOperator(pure, this, false)),
278                 array(
279                         KEYWORD(EXPORT).optional(null), //0
280                         KEYWORD(DEF), //1
281                         IDENTIFIER.between(OP(LT), OP(GT)).optional(null), //2
282                         IDENTIFIER //3
283 
284                 ).token().map(new DefinitionOperator(pure, this, true))
285 
286         );
287     }

289     private Parser<Value> dollarIdentifier(@NotNull Parser.Reference<Value> ref) {
290         return OP(DOLLAR).next(
291                 array(Terminals.Identifier.PARSER, OP(DEFAULT).next(ref.lazy()).optional(null)).between(
292                         OP(LEFT_BRACE),
293                         OP(RIGHT_BRACE)
294                 )
295         )
296                        .token()
297                        .map(token -> {
298                            Object[] objects = (Object[]) token.value();
299                            Value defaultVal = (Value) objects[1];
300                            return util().variableNode(false, VarKey.of(objects[0].toString()), false, defaultVal, token, this);
301                        });
302     }

304     private Parser<Function<? super Value, ? extends Value>> emitOperator(@NotNull Parser.Reference<Value> ref, boolean pure) {
305         return OP(EMIT_OP).token()
306                        .map(token -> {
307                            assert EMIT_OP.validForPure(pure);
308                            return lhs -> {
309                                final Value[] node = new Value[1];
310 
311                                AtomicInteger pos = new AtomicInteger(0);
312                                List<Value> values = lhs.toVarList();
313                                node[0] = util().node(EMIT_OP, pure, this, token, List.of(lhs), args -> {
314 
315                                    while (pos.get() < values.size()) {
316                                        node[0].$notify(UNARY_VALUE_CHANGE, values.get(pos.getAndIncrement()));
317                                    }
318                                    return $void();
319                                });
320 
321                                return node[0];
322                            };
323                        });
324     }

326     private Parser<Value> everyExpression(@NotNull Parser<Value> expression, boolean pure) {
327         return KEYWORD_NL(EVERY_OP)
328                        .next(array(unitExpression(false),
329                                    KEYWORD(UNTIL).next(expression).optional(null),
330                                    KEYWORD(UNLESS).next(expression).optional(null),
331                                    expression))
332                        .token()
333                        .map(token -> {
334                            assert EVERY_OP.validForPure(pure);
335                            final AtomicInteger count = new AtomicInteger(-1);
336                            Object[] objects = (Object[]) token.value();
337                            Value durationValue = (Value) objects[0];
338                            Value until = (Value) objects[1];
339                            Value unless = (Value) objects[2];
340                            Value block = (Value) objects[3];
341                            return util().reactiveNode(EVERY_OP, false, block, token, this,
342                                                       i -> everyFunc(count, durationValue, until, unless, block));
343 
344                        });
345     }

359     @Override
360     @NotNull
361     public Value parse(@NotNull Scope scope, @NotNull String source) {
362         DollarStatic.context().parser(this);
363         Value v = util().inScope(false, scope, newScope -> {
364             DollarStatic.context().classLoader(classLoader);
365             Parser<?> parser = script();
366             try {
367                 parser.from(TOKENIZER, IGNORED).parse(source);
368             } catch (RuntimeException e) {
369                 ErrorHandlerFactory.instance().handleTopLevel(e, null, (file != null) ? new File(file) : null);
370 
371             }
372             HashMap<Value, Value> exportMap = new HashMap<>();
373             exports.forEach((varKey, var) -> exportMap.put(DollarFactory.fromStringValue(varKey.asString()), var));
374             return $(exportMap);
375         }).orElseThrow(() -> new AssertionError("Optional should not be null here"));
376         if (v != null) {
377             return v;
378         } else {
379             throw new AssertionError("parse should not return null");
380         }
381 
382     }

436     @NotNull
437     private Parser<Value> expressionInternal(boolean pure, @NotNull Parser.Reference<Value> ref) {
438         Parser<Value> main;
439         if (pure) {
440             main = ref.lazy()
441                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
442                            .or(Parsers.or(unitExpression(true),
443                                           listExpression(ref.lazy(), true),
444                                           mapExpression(ref.lazy(), true),
445                                           assertExpression(ref, true),
446                                           collectExpression(ref, true),
447                                           windowExpression(ref, true),
448                                           whenExpression(ref.lazy(), true),
449                                           functionCall(true),
450                                           rangeExpression(ref, true),
451                                           variableUsageOperator(ref, true),
452                                           DECIMAL_LITERAL,
453                                           INTEGER_LITERAL,
454                                           STRING_LITERAL,
455                                           IDENTIFIER_KEYWORD,
456                                           variableRef(true),
457 
458                                           builtin(true)))
459                            .or(blockExpression(ref.lazy(), true).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
460         } else {
461             main = ref.lazy()
462                            .between(OP(LEFT_PAREN), OP(RIGHT_PAREN))
463                            .or(Parsers.or(unitExpression(false),
464                                           listExpression(ref.lazy(), false),
465                                           mapExpression(ref.lazy(), false),
466                                           KEYWORD(PURE).next(expression(true)),
467                                           classExpression(ref),
468                                           newExpression(ref, false),
469                                           thisRef(false),
470 //                                          printExpression(ref, false),
471                                           moduleExpression(ref),
472                                           assertExpression(ref, false),
473                                           collectExpression(ref, false),
474                                           windowExpression(ref, false),
475                                           whenExpression(ref.lazy(), false),
476                                           everyExpression(ref.lazy(), false),
477                                           functionCall(false),
478                                           scriptExpression(ref, false),
479                                           pureDefinitionOperator(ref),
480                                           rangeExpression(ref, false),
481                                           forkExpression(ref, false),
482                                           variableUsageOperator(ref, false),
483                                           URL,
484                                           DECIMAL_LITERAL,
485                                           INTEGER_LITERAL,
486                                           STRING_LITERAL,
487                                           dollarIdentifier(ref),
488                                           IDENTIFIER_KEYWORD,
489                                           builtin(false),
490                                           variableRef(false)))
491                            .or(blockExpression(ref.lazy(), false).between(OP_NL(LEFT_BRACE), NL_OP(RIGHT_BRACE)));
492         }
493 
494         OperatorTable<Value> table = new OperatorTable<>();
495 
496         table = infixl(pure, table, PLUS, Value::$plus);
497         table = infixl(pure, table, MINUS, Value::$minus);
498         table = infixl(pure, table, CHOOSE, Value::$choose);
499         table = infixl(pure, table, DEFAULT, Value::$default);
500         table = infixl(pure, table, DIVIDE, Value::$divide);
501         table = infixl(pure, table, MOD, Value::$modulus);
502 
503 
504         table = infixl(pure, table, INEQUALITY_OPERATOR, Value::$notEquals);
505         table = infixl(pure, table, EQUALITY, Value::$equals);
506         table = infixl(pure, table, AND, DollarStatic::$and);
507         table = infixl(pure, table, OR, DollarStatic::$or);
508         table = infixl(pure, table, LT, DollarStatic::$lt);
509         table = infixl(pure, table, GT, DollarStatic::$gt);
510         table = infixl(pure, table, LT_EQUALS, DollarStatic::$lte);
511         table = infixl(pure, table, GT_EQUALS, DollarStatic::$gte);
512         table = infixl(pure, table, MULTIPLY, Func::multiplyFunc);
513         table = infixl(pure, table, PAIR, Func::pairFunc);
514         table = infixl(pure, table, ELSE, Func::elseFunc);
515         table = infixl(pure, table, IN, Func::inFunc);
516         table = infixl(pure, table, WHEN, Func::listenFunc);
517 
518         table = infixl(pure, table, EACH, (lhs, rhs) -> eachFunc(pure, lhs, rhs));
519         table = infixl(pure, table, REDUCE, (lhs, rhs) -> reduceFunc(pure, lhs, rhs));
520         table = infixl(pure, table, CAUSES, (lhs, rhs) -> causesFunc(pure, lhs, rhs));
521 
522 
523         table = infixlReactive(pure, table, ASSERT_EQ_REACT, Func::assertEqualsFunc);
524         table = infixlUnReactive(pure, table, ASSERT_EQ_UNREACT, Func::assertEqualsFunc);
525 
526 
527         table = postfix(pure, table, DEC, (v, s) -> v.$dec());
528         table = postfix(pure, table, INC, (v, s) -> v.$inc());
529 
530         table = prefix(pure, table, NEGATE, (v, s) -> v.$negate());
531         table = prefix(pure, table, SIZE, (v, s) -> v.$size());
532 
533         table = prefix(pure, table, NOT, (v, s) -> DollarStatic.$not(v));
534         table = prefix(pure, table, ERROR, (v, s) -> Func.errorFunc(v));
535         table = prefix(pure, table, TRUTHY, (v, s) -> DollarStatic.$truthy(v));
536 
537         table = postfix(pure, table, MIN, (v, s) -> v.$min(false));
538         table = postfix(pure, table, MAX, (v, s) -> v.$max(false));
539         table = postfix(pure, table, SUM, (v, s) -> v.$sum(false));
540         table = postfix(pure, table, PRODUCT, (v, s) -> v.$product(false));
541         table = postfix(pure, table, SPLIT, (v, s) -> v.$list());
542         table = prefix(pure, table, SORT, (v, s) -> v.$sort(false));
543         table = postfix(pure, table, REVERSE, (v, s) -> v.$reverse(false));
544         table = postfix(pure, table, UNIQUE, (v, s) -> v.$unique(false));
545         table = postfix(pure, table, AVG, (v, s) -> v.$avg(false));
546 
547         table = prefixUnReactive(pure, table, FIX, (v, s) -> Func.fixFunc(v));
548 
549 //        table = table.prefix(parallelOperator(ref, pure), PARALLEL.priority());
550 //        table = table.prefix(serialOperator(ref, pure), SERIAL.priority());
551 
552         //More complex expression syntax
553         table = table.postfix(emitOperator(ref, pure), EMIT_OP.priority());
554         table = table.postfix(pipeOperator(ref, pure), PIPE_OP.priority());
555         table = table.postfix(isOperator(pure), EQ_PRIORITY);
556         table = table.postfix(memberOperator(ref, pure), MEMBER.priority());
557         table = table.postfix(subscriptOperator(ref, pure), SUBSCRIPT_OP.priority());
558         table = table.postfix(castOperator(pure), CAST.priority());
559         table = table.postfix(parameterOperator(ref, pure), PARAM_OP.priority());
560 
561         table = table.prefix(ifOperator(ref, pure), IF_OP.priority());
562         table = table.prefix(forOperator(ref, pure), FOR_OP.priority());
563         table = table.prefix(whileOperator(ref, pure), WHILE_OP.priority());
564 
565 
566         table = table.prefix(assignmentOperator(ref, pure), ASSIGNMENT.priority());
567         table = table.prefix(definitionOperator(ref, pure), DEFINITION.priority());
568 
569         if (!pure) {
570             table = infixl(false, table, PUBLISH, Func::publishFunc);
571             table = infixl(false, table, SUBSCRIBE, Func::subscribeFunc);
572             table = infixl(false, table, WRITE_SIMPLE, Func::writeFunc);
573             table = prefix(false, table, READ_SIMPLE, (v, s) -> Func.readFunc(v));
574 
575             table = prefix(false, table, DRAIN, (v, s) -> v.$drain());
576             table = prefix(false, table, ALL, (v, s) -> v.$all());
577             table = prefix(false, table, STOP, (v, s) -> v.$stop());
578             table = prefix(false, table, START, (v, s) -> v.$start());
