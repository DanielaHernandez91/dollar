67      private static String indent(int i) {
68          StringBuilder b = new StringBuilder();
69          for (int j = 0; j < i; j++) {
70              b.append(" ");
71          }
72          return b.toString();
73      }

75      private static void addScope(boolean runtime, @NotNull Scope scope) {
76          boolean newScope = scopes.get().isEmpty() || !scope.equals(currentScope());
77          scopes.get().add(scope);
78          if (DollarStatic.getConfig().debugScope()) {
79              log.info("{}{}BEGIN {}", indent(scopes.get().size() - 1), runtime ? "**** " : "", scope);
80          }
81  
82      }
83  
84      @NotNull
85      private static Scope endScope(boolean runtime) {
86  
87          Scope remove = scopes.get().remove(scopes.get().size() - 1);
88          if (DollarStatic.getConfig().debugScope()) {
89  
90              log.info("{}{}END:  {}", indent(scopes.get().size()), runtime ? "**** " : "", remove);
91  
92          }
93  
94          return remove;
95      }

229     public static <T> T inScope(boolean runtime,
230                                 @NotNull Scope parent,
231                                 boolean pure,
232                                 boolean parallel, @NotNull String scopeName,
233                                 @NotNull ScopeExecutable<T> r) {
234         Scope newScope;
235         if (pure) {
236             newScope = new PureScope(parent, parent.source(), scopeName, parent.file(), parallel);
237         } else {
238             if ((parent instanceof PureScope)) {
239                 throw new IllegalStateException(
240                                                        "trying to switch to an impure scope in a pure scope.");
241             }
242             newScope = new ScriptScope(parent, parent.file(), parent.source(), scopeName,
243                                        false, parallel);
244         }
245         addScope(runtime, parent);
246         addScope(runtime, newScope);
247         try {
248             return r.execute(newScope);
249         } catch (Exception e) {
250             newScope.handleError(e);
251             return null;
252         } finally {
253             Scope poppedScope = endScope(runtime);
254 //            poppedScope.destroy();
255             if (!Objects.equals(poppedScope, newScope)) {
256                 throw new IllegalStateException("Popped wrong scope");
257             }
258             final Scope poppedScope2 = endScope(runtime);
259             if (!Objects.equals(poppedScope2, parent)) {
260                 throw new IllegalStateException("Popped wrong scope");
261             }
262         }
263     }

267     public static <T> T inScope(boolean runtime,
268                                 @NotNull Scope scope,
269                                 @NotNull ScopeExecutable<T> r) {
270 
271         boolean addedDynamicScope = !scopes.get().isEmpty();
272 
273         addScope(runtime, scope);
274         try {
275             return r.execute(scope);
276         } catch (Exception e) {
277             scope.handleError(e);
278             return null;
279         } finally {
280             Scope poppedScope = endScope(runtime);
281             if (!Objects.equals(poppedScope, scope)) {
282                 throw new IllegalStateException("Popped wrong scope");
283             }
284 
285         }
286     }

293     @NotNull
294     public static var variableNode(boolean pure, @NotNull String key,
295                                    boolean numeric, @Nullable var defaultValue,
296                                    @NotNull Token token, @NotNull DollarParser parser) {
297         var node[] = new var[1];
298         UUID id = UUID.randomUUID();
299         node[0] = node(VAR_USAGE_OP, pure, parser, token, $(key).$list().toVarList(),
300                        (i) -> {
301                            Scope scope = currentScope();
302 
303                            if (getConfig().debugScope()) {
304                                log.info("{} {} in {} scopes ", ansiColor("LOOKUP " + key, ANSI_CYAN), scope,
305                                         scopes.get().size());
306 
307                            }
308                            if (numeric) {
309                                if (scope.hasParameter(key)) {
310                                    return scope.parameter(key);
311                                }
312                            } else {
313                                if (scope.has(key)) {
314                                    Scope scopeForKey = scope.scopeForKey(key);
315                                    assert scopeForKey != null;
316                                    log.debug("Listening to scope {} for key {}", scopeForKey, key);
317                                    scopeForKey.listen(key, id.toString(), node[0]);
318                                    Variable v = scopeForKey.variable(key);
319                                    if (!v.isPure() && (pure || scope.pure())) {
320                                        currentScope().handleError(
321                                                new PureFunctionException("Attempted to use an impure variable in a " +
322                                                                                  "pure context"));
323                                    }
324                                    return v.getValue();
325                                }
326                            }
327                            try {
328                                List<Scope> scopes = new ArrayList<>(DollarScriptSupport.scopes.get());
329                                Collections.reverse(scopes);
330                                for (Scope scriptScope : scopes) {
331                                    if (!(scriptScope instanceof PureScope) && pure) {
332                                        log.debug("Skipping {}", scriptScope);
333                                    }
334                                    if (numeric) {
335                                        if (scriptScope.hasParameter(key)) {
336                                            return scriptScope.parameter(key);
337                                        }
338                                    } else {
339                                        if (scriptScope.has(key)) {
340                                            Scope scopeForKey = scriptScope.scopeForKey(key);
341                                            assert scopeForKey != null;
342                                            log.debug("Listening to scope {} for key {}", scopeForKey, key);
343                                            scopeForKey.listen(key, id.toString(), node[0]);
344                                            Variable v = scopeForKey.variable(key);
345                                            if (!v.isPure() && (pure || scope.pure())) {
346                                                scope.handleError(
347                                                        new PureFunctionException("Attempted to use an impure " +
348                                                                                          "variable in a pure " +
349                                                                                          "context"));
350                                            }
351                                            return v.getValue();
352                                        }
353                                    }
354                                }
355                            } catch (DollarAssertionException e) {
356                                throw e;
357                            } catch (DollarScriptException e) {
358                                return ErrorHandlerFactory.instance().handle(scope, null, e);
359                            } catch (RuntimeException e) {
360                                return ErrorHandlerFactory.instance().handle(scope, null, e);
361                            }
362                            if (numeric) {
363                                throw new VariableNotFoundException(key, scope);
364                            }
365 
366                            if (defaultValue != null) {
367                                return defaultValue;
368                            } else {
369                                throw new VariableNotFoundException(key, scope);
370                            }
371                        }
372         );
373         node[0].metaAttribute(VARIABLE, key);
374         return node[0];
375 
376     }

379     @Nullable
380     public static Scope getScopeForVar(boolean pure,
381                                        @NotNull String key,
382                                        boolean numeric,
383                                        @Nullable Scope initialScope) {
384 
385         if (initialScope == null) {
386             initialScope = currentScope();
387         }
388         if (getConfig().debugScope()) {
389             log.info("{} {} in {} scopes ", ansiColor("LOOKUP " + key, ANSI_CYAN), initialScope, scopes.get().size());
390         }
391         if (numeric) {
392             if (initialScope.hasParameter(key)) {
393                 return initialScope;
394             }
395         } else {
396             if (initialScope.has(key)) {
397                 return initialScope;
398             }
399         }
400 
401         List<Scope> scopes = new ArrayList<>(DollarScriptSupport.scopes.get());
402         Collections.reverse(scopes);
403         for (Scope scriptScope : scopes) {
404             if (!(scriptScope instanceof PureScope) && pure) {
405                 log.debug("Skipping {}", scriptScope);
406             }
407             if (numeric) {
408                 if (scriptScope.hasParameter(key)) {
409                     return scriptScope;
410                 }
411             } else {
412                 if (scriptScope.has(key)) {
413 
414                     return scriptScope;
415                 }
416             }
417         }
418 
419         return null;
420

448     @NotNull
449     public static var setVariable(@NotNull Scope scope,
450                                   @NotNull String key,
451                                   @NotNull var value,
452                                   boolean readonly,
453                                   @Nullable var useConstraint,
454                                   @Nullable String useSource,
455                                   boolean isVolatile,
456                                   boolean fixed,
457                                   boolean pure,
458                                   boolean decleration,
459                                   @NotNull Token token,
460                                   @Nullable DollarParser parser) {
461 
462         SourceSegment source = new SourceSegmentValue(scope, token);
463         boolean numeric = key.matches("^\\d+$");
464 
465 
466         if (scope.has(key)) {
467             return updateVariable(scope, key, value, readonly, useConstraint, useSource,
468                                   isVolatile, fixed, pure, decleration);
469         }
470         try {
471             List<Scope> scopes = new ArrayList<>(DollarScriptSupport.scopes.get());
472             Collections.reverse(scopes);
473             for (Scope scriptScope : scopes) {
474                 if (!(scriptScope instanceof PureScope) && pure) {
475                     log.debug("Skipping {}", scriptScope);
476                 }
477 
478                 if (scriptScope.has(key)) {
479                     return updateVariable(scriptScope, key, value, readonly, useConstraint, useSource,
480                                           isVolatile, fixed, pure, decleration);
481                 }
482 
483             }
484         } catch (DollarAssertionException e) {
485             return ErrorHandlerFactory.instance().handle(scope, null, e);
486         } catch (DollarScriptException e) {
487             return ErrorHandlerFactory.instance().handle(scope, null, e);
488         } catch (RuntimeException e) {
489             return ErrorHandlerFactory.instance().handle(scope, null, e);
490         }
491         if (numeric) {
492             return scope.parameter(key);
493         }
494 
495         if (decleration) {
496             if (getConfig().debugScope()) {
497                 log.info("{} {} {}", ansiColor("SETTING  " + key, ANSI_CYAN), scope, scope);
498             }
499             return scope.set(key, value, readonly, useConstraint, useSource, isVolatile,
500                              fixed,
501                              pure);
502         } else {
503             throw new VariableNotFoundException(key, scope);
504         }
505

517     private static var updateVariable(@NotNull Scope scope,
518                                       @NotNull String key,
519                                       @NotNull var value,
520                                       boolean readonly,
521                                       @Nullable var useConstraint,
522                                       @Nullable String useSource,
523                                       boolean isVolatile,
524                                       boolean fixed,
525                                       boolean pure, boolean decleration) {
526         if (getConfig().debugScope()) {
527             log.info("{}{} {}", ansiColor("UPDATING ", ANSI_CYAN), key, scope);
528         }
529         if (decleration) {
530             throw new DollarScriptException("Variable " + key + " already defined in " + scope);
531         } else {
532 
533             return scope.set(key, value, readonly, useConstraint, useSource, isVolatile, fixed, pure);
534         }
535     }

537     @NotNull
538     public static var constrain(@NotNull Scope scope, @NotNull var value, @Nullable var constraint, @Nullable String source) {
539 //        System.err.println("(" + source + ") " + rhs.$type().constraint());
540         if (!Objects.equals(value.constraintLabel(), source)) {
541             if ((source != null) && (value.constraintLabel() != null) && !value.constraintLabel().isEmpty()) {
542                 scope.handleError(
543                         new DollarScriptException("Trying to assign an invalid constrained variable " + value.constraintLabel() + " vs " + source,
544                                                   value));
545             }
546         } else {
547             if ((value.constraintLabel() != null) && !value.constraintLabel().isEmpty()) {
548 //                System.err.println("Fingerprint: " + rhs.$type().constraint());
549             }
550         }
551         return value.$constrain(constraint, source);
552     }

554     public static String ansiColor(@NotNull String text, @NotNull String color) {
555         return "\u001b["  // Prefix
556                        + "0"        // Brightness
557                        + ";"        // Separator
558                        + color       // Red foreground
559                        + "m"        // Suffix
560                        + text       // the text to output
561                        + "\u001b[m"; // Prefix + Suffix to reset color
562     }
