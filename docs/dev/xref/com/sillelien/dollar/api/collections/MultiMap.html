17  package com.sillelien.dollar.api.collections;
18  
19  import org.jetbrains.annotations.NotNull;
20  import org.jetbrains.annotations.Nullable;
21  
22  import java.io.Serializable;
23  import java.util.Collection;
24  import java.util.Iterator;
25  import java.util.Map;
26  
27  public interface MultiMap<K, V> extends Cloneable, Serializable {
28  
29      @NotNull
30      Collection<V> allValues();
31  
32      /**
33       * Clear the map. <p> This clears each collection in the map, and so may be slow.
34       */
35      void clear();
36  
37      boolean containsKey(@NotNull Object key);
38  
39      /**
40       * Checks whether the map contains the value specified. <p> This checks all collections against all keys for the
41       * value, and thus could be slow.
42       *
43       * @param value the value to search for
44       * @return true if the map contains the value
45       */
46      boolean containsValue(@NotNull Object value);
47  
48      /**
49       * Checks whether the collection at the specified key contains the value.
50       *
51       * @param value the value to search for
52       * @return true if the map contains the value
53       * @since Commons Collections 3.1
54       */
55      boolean containsValue(@NotNull Object key, @NotNull Object value);
56  
57      @NotNull
58      Iterable<? extends Map.Entry<K, Collection<V>>> entries();
59  
60      /**
61       * Gets the collection mapped to the specified key. This method is a convenience method to typecast the result of
62       * {@code get(key)}.
63       *
64       * @param key the key to retrieve
65       * @return the collection mapped to the key, null if no mapping
66       * @since Commons Collections 3.1
67       */
68      @NotNull
69      Collection<V> getCollection(@NotNull K key);
70  
71      /**
72       * Gets an iterator for the collection mapped to the specified key.
73       *
74       * @param key the key to get an iterator for
75       * @return the iterator of the collection at the key, empty iterator if key not in map
76       * @since Commons Collections 3.1
77       */
78      @Nullable
79      Iterator<V> iterator(@NotNull Object key);
80  
81      /**
82       * Adds a collection of values to the collection associated with the specified key.
83       *
84       * @param key    the key to store against
85       * @param values the values to add to the collection at the key, null ignored
86       * @return true if this map changed
87       * @since Commons Collections 3.1
88       */
89      boolean putAll(@NotNull Object key, @NotNull Collection<V> values);
90  
91      /**
92       * Adds the value to the collection associated with the specified key. <p> Unlike a normal {@code Map} the
93       * previous value is not replaced. Instead the new value is added to the collection stored against the key.
94       *
95       * @param key   the key to store against
96       * @param value the value to add to the collection at the key
97       * @return the value added if the map changed and null if the map did not change
98       */
99      @Nullable
100     V putValue(@NotNull K key, @NotNull V value);
101 
102     /**
103      * Removes a specific value from map. <p> The item is removed from the collection mapped to the specified key. Other
104      * values attached to that key are unaffected. <p> If the last value for a key is removed, {@code null} will be
105      * returned from a subsequant {@code get(key)}.
106      *
107      * @param key  the key to remove from
108      * @param item the value to remove
109      * @return the value removed (which was passed in), null if nothing removed
110      */
111     boolean remove(@NotNull Object key, @NotNull Object item);
112 
113     /**
114      * Gets the size of the collection mapped to the specified key.
115      *
116      * @param key the key to get size for
117      * @return the size of the collection at the key, zero if key not in map
118      * @since Commons Collections 3.1
119      */
120     int size(@NotNull Object key);
121 
122     /**
123      * Gets a collection containing all the values in the map. <p> This returns a collection containing the combination
124      * of values from all keys.
125      *
126      * @return a collection view of the values contained in this map
127      */
128     @NotNull
129     Collection<Collection<V>> values();
130 
131 }
